  7 define(["DomUtils"],
  8 
  9 /**
 10 * @class
 11 * Event plugin adds events listeners to DOM nodes.
 12 * It can also delegate the event handling to a parent dom node
 13 * @requires Utils
 14 */
 15 function EventPlugin(Utils) {
 16 
 17 	/**
 18 	 * The event plugin constructor.
 19 	 * ex: new EventPlugin({method: function(){} ...}, false);
 20 	 * @param {Object} the object that has the event handling methods
 21 	 * @param {Boolean} $isMobile if the event handler has to map with touch events
 22 	 */
 23 	return function EventPluginConstructor($parent, $isMobile) {
 24 
 25 		/**
 26 		 * The parent callback
 27 		 * @private
 28 		 */
 29 		var _parent = null,
 30 
 31 		/**
 32 		 * The mapping object.
 33 		 * @private
 34 		 */
 35 		_map = {
 36 			"mousedown" : "touchstart",
 37 			"mouseup" : "touchend",
 38 			"mousemove" : "touchmove"
 39 		},
 40 
 41 		/**
 42 		 * Is touch device.
 43 		 * @private
 44 		 */
 45 		_isMobile = !!$isMobile;
 46 
 47 		/**
 48 		 * Add mapped event listener (for testing purpose).
 49 		 * @private
 50 		 */
 51 		this.addEventListener = function addEventListener(node, event, callback, useCapture) {
 52 			node.addEventListener(this.map(event), callback, !!useCapture);
 53 		};
 54 
 55 		/**
 56 		 * Listen to DOM events.
 57 		 * @param {Object} node DOM node
 58 		 * @param {String} name event's name
 59 		 * @param {String} listener callback's name
 60 		 * @param {String} useCapture string
 61 		 */
 62 		this.listen = function listen(node, name, listener, useCapture) {
 63 			this.addEventListener(node, name, function(e){
 64 				_parent[listener].call(_parent, e, node);
 65 			}, !!useCapture);
 66 		};
 67 
 68 		/**
 69 		 * Delegate the event handling to a parent DOM element
 70 		 * @param {Object} node DOM node
 71 		 * @param {String} selector CSS3 selector to the element that listens to the event
 72 		 * @param {String} name event's name
 73 		 * @param {String} listener callback's name
 74 		 * @param {String} useCapture string
 75 		 */
 76 		this.delegate = function delegate(node, selector, name, listener, useCapture) {
 77 			this.addEventListener(node, name, function(event){
 78 				if (Utils.matches(node, selector, event.target)) {
 79 					_parent[listener].call(_parent, event, node);
 80 				}
 81 			}, !!useCapture);
 82 		};
 83 
 84 		/**
 85 		 * Get the parent object.
 86 		 * @return {Object} the parent object
 87 		 */
 88 		this.getParent = function getParent() {
 89 			return _parent;
 90 		};
 91 
 92 		/**
 93 		 * Set the parent object.
 94 		 * The parent object is an object which the functions are called by node listeners.
 95 		 * @param {Object} the parent object
 96 		 * @return true if object has been set
 97 		 */
 98 		this.setParent = function setParent(parent) {
 99 			if (parent instanceof Object){
100 				_parent = parent;
101 				return true;
102 			}
103 			return false;
104 		};
105 
106 		/**
107 		 * Get event mapping.
108 		 * @param {String} event's name
109 		 * @return the mapped event's name
110 		 */
111 		this.map = function map(name) {
112 			return _isMobile ? (_map[name] || name) : name;
113 		};
114 
115 		/**
116 		 * Set event mapping.
117 		 * @param {String} event's name
118 		 * @param {String} event's value
119 		 * @return true if mapped
120 		 */
121 		this.setMap = function setMap(name, value) {
122 			if (typeof name == "string" &&
123 				typeof value == "string") {
124 				_map[name] = value;
125 				return true;
126 			}
127 			return false;
128 		};
129 
130 		//init
131 		this.setParent($parent);
132 	};
133 
134 });
