  7 define(["OObject", "Tools"],
  8 /**
  9 * @class
 10 * Place plugin places OObject in the DOM.
 11 * @requires OObject, Tools
 12 */
 13 function PlacePlugin(OObject, Tools) {
 14 
 15 	/**
 16 	 * Intilialize a Place.plugin with a list of OObjects
 17 	 * @param {Object} $uis a list of OObjects such as:
 18 	 *   {
 19 	 *		"header": new OObject(),
 20 	 *		"list": new OObject()
 21 	 *	 }
 22 	 * @Constructor
 23 	 */
 24 	return function PlacePluginConstructor($uis) {
 25 
 26 		/**
 27 		 * The list of uis currently set in this place plugin
 28 		 * @private
 29 		 */
 30 		var _uis = {};
 31 
 32 		/**
 33 		 * Attach an OObject to this DOM element
 34 		 * @param {HTML|SVGElement} node the dom node where to attach the OObject
 35 		 * @param {String} the name of the OObject to attach
 36 		 * @throws {NoSuchOObject} an error if there's no OObject for the given name
 37 		 */
 38 		this.place = function place(node, name) {
 39 			if (_uis[name] instanceof OObject) {
 40 				_uis[name].place(node);
 41 			} else {
 42 				throw new Error(name + " is not an OObject UI in place:"+name);
 43 			}
 44 		};
 45 
 46 		/**
 47 		 * Add an OObject that can be attached to a dom element
 48 		 * @param {String} the name of the OObject to add to the list
 49 		 * @param {OObject} ui the OObject to add the list
 50 		 * @returns {Boolean} true if the OObject was added
 51 		 */
 52 		this.set = function set(name, ui) {
 53 			if (typeof name == "string" && ui instanceof OObject) {
 54 				_uis[name] = ui;
 55 				return true;
 56 			} else {
 57 				return false;
 58 			}
 59 		};
 60 
 61 		/**
 62 		 * Add multiple dom elements at once
 63 		 * @param {Object} $uis a list of OObjects such as:
 64 		 *   {
 65 		 *		"header": new OObject(),
 66 		 *		"list": new OObject()
 67 		 *	 }
 68 		 */
 69 		this.setAll = function setAll(uis) {
 70 			Tools.loop(uis, function (ui, name) {
 71 				this.set(name, ui);
 72 			}, this);
 73 		};
 74 
 75 		/**
 76 		 * Returns an OObject from the list given its name
 77 		 * @param {String} the name of the OObject to get
 78 		 * @returns {OObject} OObject for the given name
 79 		 */
 80 		this.get = function get(name) {
 81 			return _uis[name];
 82 		};
 83 
 84 		this.setAll($uis);
 85 
 86 	};
 87 
 88 });
