  7 define(["OObject", "Tools"],
  8 
  9 function (OObject, Tools) {
 10 
 11 	return function PlacePluginConstructor($uis) {
 12 
 13 		var _uis = {};
 14 
 15 		this.place = function place(node, name) {
 16 			if (_uis[name] instanceof OObject) {
 17 				_uis[name].place(node);
 18 			} else {
 19 				throw new Error(name + " is not an OObject UI in place:"+name);
 20 			}
 21 		};
 22 
 23 		this.set = function set(name, ui) {
 24 			if (typeof name == "string" && ui instanceof OObject) {
 25 				_uis[name] = ui;
 26 				return true;
 27 			} else {
 28 				return false;
 29 			}
 30 		};
 31 
 32 		this.setAll = function setAll(uis) {
 33 			Tools.loop(uis, function (ui, name) {
 34 				this.set(name, ui);
 35 			}, this);
 36 		};
 37 
 38 		this.get = function get(name) {
 39 			return _uis[name];
 40 		};
 41 
 42 		this.setAll($uis);
 43 
 44 	};
 45 
 46 });
