1 define("Olives/Text",
  2 	
  3 ["Olives/OObject"],
  4 
  5 /** 
  6 * @class 
  7 * Text is a simple paragraph with a unique property that shows the way UI work.
  8 */		
  9 function Text(OObject) {
 10 	
 11 	function TextConstructor(texts) {
 12 		
 13 		/**
 14 		 * Set the paragraph's innherHTML with given content
 15 		 * @param {String} content the text
 16 		 */
 17 		var setInnerHTML = function setInnerHTML(node, value) {
 18 			node.innerHTML = value;		
 19 		};
 20 		
 21 		/**
 22 		 * Construction of the UI
 23 		 */
 24 		// Sets the data
 25 		this.model.reset(texts);
 26 		// Watch for modifications
 27 		this.model.watch("updated", function (name, value) {
 28 			setInnerHTML(this.connects[name], value);
 29 		}, this);
 30 		
 31 		/**
 32 		 * The UI's template that will generate the dom nodes to be placed on the application's page.
 33 		 */
 34 		this.template = "<p data-connect='content'></p>";
 35 
 36 		/**
 37 		 * onRender is called after action("render")
 38 		 */
 39 		this.onRender = function onRender() {
 40 			for (var name in this.connects) {
 41 				if (this.model.has(name)) {
 42 					setInnerHTML(this.connects[name], this.model.get(name));
 43 				} else {
 44 					this.model.set(name, this.connects[name].innerHTML);
 45 				}
 46 
 47 			}
 48 		};
 49 		
 50 	}
 51 	
 52 	return function TextFactory(texts) {
 53 		TextConstructor.prototype = new OObject;
 54 		return new TextConstructor(texts);
 55 	};
 56 
 57 	
 58 });
