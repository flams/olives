1 /**
  2  * @class
  3  * Olives is the UI loader
  4  * It's based on Emily
  5  * You can define UIElements and create new instances of them.
  6  */
  7 function Olives(FWK) {
  8 	
  9 	/**
 10 	 * Olives' namespace in Emily
 11 	 * @private
 12 	 */
 13 	var _namespace = "Olives",
 14 	
 15 	/**
 16 	 * Builds up the namespace
 17 	 * @private
 18 	 * @param {String} name the name of the UI
 19 	 * @returns {String} the UI's namespace
 20 	 */
 21 	_addNS = function (name) {
 22 		return _namespace + "." + name;
 23 	};
 24 		
 25 	/**
 26 	 * defines a new UI
 27 	 * @param {String} name the name of the class
 28 	 * @param {String} inherits the name of the class it inherits from (optional)
 29 	 * @param {Function} constructor the function constructor
 30 	 * @returns {Function} the function constructor
 31 	 */
 32 	this.define = function define(name, inherits, constructor) {
 33 		inherits = typeof inherits == "string" ? _addNS(inherits) : inherits;
 34 		return FWK.declare(_addNS(name), inherits, constructor);
 35 	};
 36 		
 37 	/**
 38 	 * creates a new instance of a previously declared class
 39 	 * @param {String} name the name of the UI to create
 40 	 * @returns {Object} the new UI instance
 41 	 */
 42 	this.create = function create(name) {
 43 		return Object.create(FWK.require(_addNS(name)));
 44 	};
 45 };
 46 
 47 var Olives = new Olives(Emily);
