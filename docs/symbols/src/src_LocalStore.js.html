  7 define("Olives/LocalStore", 
  8 		
  9 ["Store", "Tools"],
 10 		
 11 /**
 12  * @class
 13  * LocalStore is an Emily's Store that can be synchronized with localStorage
 14  * Synchronize the store, reload your page/browser and resynchronize it with the same value
 15  * and it gets restored.
 16  * Only valid JSON data will be stored
 17  */
 18 function LocalStore(Store, Tools) {
 19 	
 20 	function LocalStoreConstructor() {
 21 		
 22 		/**
 23 		 * The name of the property in which to store the data 
 24 		 * @private
 25 		 */
 26 		var _name = null,
 27 		
 28 		/**
 29 		 * Saves the current values in localStorage
 30 		 * @private
 31 		 */
 32 		setLocalStorage = function () {
 33 			localStorage.setItem(_name, this.toJSON());
 34 		};
 35 		
 36 		/**
 37 		 * Synchronize the store with localStorage
 38 		 * @param {String} name the name in which to save the data
 39 		 * @returns {Boolean} true if the param is a string
 40 		 */
 41 		this.sync = function sync(name) {
 42 			if (typeof name == "string") {
 43 				_name = name;
 44 				Tools.loop(JSON.parse(localStorage.getItem(name)), function (value, idx) {
 45 					if (!this.has(idx)) {
 46 						this.set(idx, value);
 47 					}
 48 				}, this);
 49 				setLocalStorage.call(this);
 50 				return true;
 51 			} else {
 52 				return false;
 53 			}
 54 		};
 55 		
 56 		// Watch for modifications to update localStorage
 57 		this.watch("added", setLocalStorage, this);
 58 		this.watch("updated", setLocalStorage, this);
 59 		this.watch("deleted", setLocalStorage, this);
 60 		
 61 	}
 62 	
 63 	return function LocalStoreFactory() {
 64 		LocalStoreConstructor.prototype = new Store;
 65 		return new LocalStoreConstructor;
 66 	};
 67 	
 68 });
