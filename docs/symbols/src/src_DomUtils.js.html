 10 		/**
 11 		 * Returns a NodeList including the given dom node,
 12 		 * its childNodes and its siblingNodes
 13 		 * @param {HTMLElement} dom the dom node to start with
 14 		 * @param {String} query an optional CSS selector to narrow down the query
 15 		 * @returns the list of nodes
 16 		 */
 17 		getNodes: function getNodes(dom, query) {
 18 			if (dom instanceof HTMLElement) {
 19 				if (!dom.parentNode) {
 20 					document.createDocumentFragment().appendChild(dom);
 21 				}
 22 
 23 				return dom.parentNode.querySelectorAll(query || "*");
 24 			} else {
 25 				return false;
 26 			}
 27 		},
 28 	
 29 		getDataset: function getDataset(dom) {
 30 			var i=0,
 31 				l, 
 32 				dataset={},
 33 				split,
 34 				join;
 35 			
 36 			if (dom instanceof HTMLElement) {
 37 				for (l=dom.attributes.length;i<l;i++) {
 38 					split = dom.attributes[i].name.split("-");
 39 					if (split.shift() == "data") {
 40 						dataset[join = split.join("-")] = dom.getAttribute("data-"+join);
 41 					}
 42 				}
 43 				return dataset;
 44 			} else {
 45 				return false;
 46 			}
 47 		}
