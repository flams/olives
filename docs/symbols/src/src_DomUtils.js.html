  7 define("Olives/DomUtils", function () {
  8 
  9 	return {
 10 		/**
 11 		 * Returns a NodeList including the given dom node,
 12 		 * its childNodes and its siblingNodes
 13 		 * @param {HTMLElement|SVGElement} dom the dom node to start with
 14 		 * @param {String} query an optional CSS selector to narrow down the query
 15 		 * @returns the list of nodes
 16 		 */
 17 		getNodes: function getNodes(dom, query) {
 18 			if (this.isAcceptedType(dom)) {
 19 				if (!dom.parentNode) {
 20 					document.createDocumentFragment().appendChild(dom);
 21 				}
 22 
 23 				return dom.parentNode.querySelectorAll(query || "*");
 24 			} else {
 25 				return false;
 26 			}
 27 		},
 28 	
 29 		/**
 30 		 * Get a domNode's dataset attribute. If dataset doesn't exist (IE) 
 31 		 * then the domNode is looped through to collect them.
 32 		 * @param {HTMLElement|SVGElement} dom
 33 		 * @returns {Object} dataset
 34 		 */
 35 		getDataset: function getDataset(dom) {
 36 			var i=0,
 37 				l, 
 38 				dataset={},
 39 				split,
 40 				join;
 41 			
 42 			if (this.isAcceptedType(dom)) {
 43 				if (dom.hasOwnProperty("dataset")) {
 44 					return dom.dataset;
 45 				} else {
 46 					for (l=dom.attributes.length;i<l;i++) {
 47 						split = dom.attributes[i].name.split("-");
 48 						if (split.shift() == "data") {
 49 							dataset[join = split.join("-")] = dom.getAttribute("data-"+join);
 50 						}
 51 					}
 52 					return dataset;
 53 				}
 54 				
 55 			} else {
 56 				return false;
 57 			}
 58 		},
 59 		
 60 		/**
 61 		 * Olives can manipulate HTMLElement and SVGElements
 62 		 * This function tells if an element is one of them
 63 		 * @param {Element} type
 64 		 * @returns true if HTMLElement or SVGElement
 65 		 */
 66 		isAcceptedType: function isAcceptedType(type) {
 67 			if (type instanceof HTMLElement ||
 68 				type instanceof SVGElement) {
 69 				return true;
 70 			} else {
 71 				return false;
 72 			}
 73 		},
 74 		
 75 		/**
 76 		 * Assign a new value to an Element's property. Works with HTMLElement and SVGElement.
 77 		 * @param {HTMLElement|SVGElement} node the node which property should be changed
 78 		 * @param {String} property the name of the property
 79 		 * @param {any} value the value to set
 80 		 * @returns true if assigned
 81 		 */
 82 		setAttribute: function setAttribute(node, property, value) {
 83 				if (node instanceof HTMLElement) {
 84 					node[property] = value;
 85 					return true;
 86 				} else if (node instanceof SVGElement){
 87 					node.setAttribute(property, value);
 88 					return true;
 89 				} else {
 90 					return false;
 91 				}
 92 		}
 93 	
 94 	};
 95 
 96 });
