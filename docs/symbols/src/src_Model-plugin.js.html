1 define("Olives/Model-plugin", 
  2 		
  3 ["Store", "Observable", "Tools", "Olives/DomUtils"],
  4 		
  5 /**
  6  * @class
  7  * This plugin links dom nodes to a model
  8  * @requires Store, Observable
  9  */
 10 function ModelPlugin(Store, Observable, Tools, DomUtils) {
 11 	
 12 	return function ModelPluginConstructor($model) {
 13 		
 14 		/**
 15 		 * The model to watch
 16 		 * @private
 17 		 */
 18 		var _model = null,
 19 		
 20 		/**
 21 		 * An observable that apdapts to model's observable
 22 		 * @private
 23 		 */
 24 		_observable = new Observable();
 25 		
 26 		/**
 27 		 * Define the model to watch for
 28 		 * @param {Store} model the model to watch for changes
 29 		 * @returns {Boolean} true if the model was set
 30 		 */
 31 		this.setModel = function setModel(model) {
 32 			if (model instanceof Store) {
 33 				// Set the model
 34 				_model = model;
 35 				// Also watch for value modification
 36 				_model.watch("updated", function (name, value) {
 37 					_observable.notify(name, value);
 38 				});
 39 				_model.watch("added", function (name, value) {
 40 					_observable.notify(name, value);
 41 				});
 42 				return true;
 43 			} else {
 44 				return false;
 45 			}
 46 		};
 47 		
 48 		/**
 49 		 * Get the store that is watched for
 50 		 * @returns the Store
 51 		 */
 52 		this.getModel = function getModel() {
 53 			return _model;
 54 		};
 55 		
 56 		/**
 57 		 * The item renderer defines a dom node that can be duplicated
 58 		 * It is made available for debugging purpose, don't use it
 59 		 * @private
 60 		 */
 61 		this.ItemRenderer = function ItemRenderer(item) {
 62 			
 63 			var _node = null;
 64 			
 65 			/**
 66 			 * Set the duplicated node
 67 			 * @private
 68 			 */
 69 			this.set = function set(node) {
 70 				_node = node;
 71 				return true;
 72 			};
 73 			
 74 			/**
 75 			 * @private
 76 			 * @returns the node that is duplicated
 77 			 */
 78 			this.get = function get() {
 79 				return _node;
 80 			};
 81 			
 82 			/**
 83 			 * Associate the duplicated node to an item in the model
 84 			 * @param id
 85 			 * @param pluginName
 86 			 * @returns the associated node
 87 			 */
 88 			this.associate = function associate(id, pluginName, hey) {
 89 				var newNode = _node.cloneNode(true);
 90 				var nodes = DomUtils.getNodes(newNode, "[data-" + pluginName + "]");
 91 
 92 				
 93 				Tools.loop(nodes, function (child) {
 94 						// Emily's Tools.loop returns me 0 on empty NodeList
 95 						// Must fix this in the function instead of here
 96             			child && (child.dataset[pluginName+".id"] = id);
 97 				});
 98 				return newNode;
 99 			};
100 			
101 			this.set(item);			
102 		};
103 		
104 		/**
105 		 * Expands the inner dom nodes of a given dom node, filling it with model's values
106 		 * @param {HTMLElement} node the dom node to apply toList to
107 		 */
108 		this.toList = function toList(node) {
109 			var domFragment,
110 				itemRenderer = new this.ItemRenderer(node.childNodes[0]);
111 
112 	
113 			domFragment = document.createDocumentFragment();
114             _model.loop(function (value, idx) {
115                     domFragment.appendChild(this.plugins.apply(itemRenderer.associate(idx, this.plugins.name)));
116             }, this);
117             
118 
119 	         node.replaceChild(domFragment, node.childNodes[0]);
120             
121             _model.watch("added", function (idx, value) {
122                 node.insertBefore(this.plugins.apply(itemRenderer.associate(idx, this.plugins.name)), node.childNodes[idx]);
123             }, this);
124             
125             _model.watch("deleted", function (idx) {
126                     // The document.childNodes indexes need to be preserved,
127                     // so I replace the nodes with empty ones.
128                     node.replaceChild(document.createTextNode(""), node.childNodes[idx]);
129             });
130 
131 
132          };
133 		
134 		/**
135 		 * Attach a model's value to a dom node so it also gets updated on value's changes
136 		 * @param {HTMLElement} node the dom node to apply the plugin to
137 		 * @param {String} name the name of the property to look for in the model's value
138 		 * @returns
139 		 */
140 		this.toText = function toText(node, name) {
141 			
142 			// Name can be unset if the value of a row is plain text
143 			name = name || "";
144 
145 			// In case of an array-like model the id is the index of the model's item to look for.
146 			// The .id is added by the toList function
147 			var id = node.dataset[this.plugins.name+".id"],
148 			
149 			// Else, it is the first element of the following
150 			split = name.split("."),
151 			
152 			// So the index of the model is either id or the first element of split
153 			modelIdx = id || split.shift(),
154 			
155 			// And the name of the property to look for in the value is
156 			prop = id ? name : split.join("."),
157 			
158 			// Get the model's value
159 			get =  Tools.getObjectsProperty(_model.get(modelIdx), prop);
160 			
161 			// If the value is falsy, the dom shouldn't be touched
162 			if (get) {
163 				node.innerHTML = get;
164 			}
165 			
166 			// Watch for changes
167 			_observable.watch(modelIdx, function (value) {
168 					node.innerHTML = Tools.getObjectsProperty(value, prop);
169 			});
170 			
171 			
172 		};
173 	
174 		// Inits the model
175 		this.setModel($model);
176 		
177 		
178 	};
179 	
180 });
