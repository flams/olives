1 define("Olives/Model-plugin", 
  2 		
  3 ["Store", "Observable", "Tools"],
  4 		
  5 /**
  6  * @class
  7  * This plugin links dom nodes to a model
  8  * @requires Store, Observable
  9  */
 10 function ModelPlugin(Store, Observable, Tools) {
 11 	
 12 	return function ModelPluginConstructor($model) {
 13 		
 14 		/**
 15 		 * The model to watch
 16 		 * @private
 17 		 */
 18 		var _model = null,
 19 		
 20 		/**
 21 		 * An observable that apdapts to model's observable
 22 		 * @private
 23 		 */
 24 		_observable = new Observable(),
 25 		
 26 		/**
 27 		 * The function to put some content into a dom node
 28 		 * @private
 29 		 */
 30 		setInnerHTML = function setInnerHTML(dom, html) {
 31 			dom.innerHTML = html;
 32 		};
 33 		
 34 		/**
 35 		 * Define the model to watch for
 36 		 * @param {Store} model the model to watch for changes
 37 		 * @returns {Boolean} true if the model was set
 38 		 */
 39 		this.setModel = function setModel(model) {
 40 			if (model instanceof Store) {
 41 				// Set the model
 42 				_model = model;
 43 				// Also watch for value modification
 44 				_model.watch("updated", function (name, value) {
 45 					_observable.notify(name, value);
 46 				});
 47 				_model.watch("added", function (name, value) {
 48 					_observable.notify(name, value);
 49 				});
 50 				return true;
 51 			} else {
 52 				return false;
 53 			}
 54 		};
 55 		
 56 		/**
 57 		 * Get the store that is watched for
 58 		 * @returns the Store
 59 		 */
 60 		this.getModel = function getModel() {
 61 			return _model;
 62 		};
 63 		
 64 		/**
 65 		 * 
 66 		 * 
 67 		 * 
 68 		 * 
 69 		 * 
 70 		 * 
 71 		 * 
 72 		 * 
 73 		 * 
 74 		 * 
 75 		 * 
 76 		 * 
 77 		 * 
 78 		 * Refactoring ongoing
 79 		 * 
 80 		 * 
 81 		 * 
 82 		 * 
 83 		 * 
 84 		 * 
 85 		 * 
 86 		 * 
 87 		 * 
 88 		 * 
 89 		 */
 90 		
 91 		/**
 92 		 * ...
 93 		 * @param {HTMLElement} node the dom node to apply toList to
 94 		 */
 95 		this.toList = function toList(node) {
 96 			var itemRenderer = node.childNodes[0],
 97             	domFragment = document.createDocumentFragment(),
 98             	pluginName = this.getName();
 99 
100 			
101 			if (itemRenderer) {
102 
103 	            _model.loop(function (value, idx) {
104 	                    var newNode = itemRenderer.cloneNode(true);
105 	                    if (newNode.childNodes) {
106 	                    	Tools.loop(newNode.childNodes, function (child) {
107 	                    		if (child.dataset[pluginName]) {
108 	                    			child.dataset[pluginName+".id"] = idx;
109 	                    		}
110 	                    	});
111 	                    }
112 	                    if(newNode.dataset[pluginName]) {
113 	                    	newNode.dataset[pluginName+".id"] = idx;
114 	                    }
115 	                    this.apply(newNode);
116 	                    domFragment.appendChild(newNode);
117 	            }, this);
118 
119                
120 	            
121 	            node.replaceChild(domFragment, itemRenderer);
122             
123 			}
124             
125             _model.watch("added", function (idx, value) {
126                     var newNode = itemRenderer.cloneNode(true);
127                     if (newNode.childNodes) {
128                     	Tools.loop(newNode.childNodes, function (child) {
129                     		if (child.dataset[pluginName]) {
130                     			child.dataset[pluginName+".id"] = idx;
131                     		}
132                     	});
133                     }
134                     if(newNode.dataset[pluginName]) {
135                     	newNode.dataset[pluginName+".id"] = idx;
136                     }
137                     
138                     this.apply(newNode);
139                     
140                     node.insertBefore(newNode, node.childNodes[idx]);
141             }, this);
142             
143             _model.watch("deleted", function (idx) {
144                     // The document.childNodes indexes need to be preserved,
145                     // so I replace the nodes with empty ones.
146                     node.replaceChild(document.createTextNode(""), node.childNodes[idx]);
147             });
148 
149          };
150 		
151 		
152 		/**
153 		 * Attach a model's value to a dom node so it also gets updated on value's changes
154 		 * @param {HTMLElement} node the dom node to apply the plugin to
155 		 * @param {String} name the name of the model's value to attach it to
156 		 * @returns
157 		 */
158 		this.toText = function toText(node, name) {
159 			// Leave the dom intact of no value
160 			if (_model.has(name)) {
161 				setInnerHTML(node, _model.get(name));
162 			}
163 			// Watch for modifications
164 			_observable.watch(name, function (value) {
165 				setInnerHTML(node, value);
166 			});
167 		};
168 		
169 		this.item = function item(node, prop) {
170 			var id = node.dataset[this.getName()+".id"];
171 
172 			// Leave the dom intact of no value
173 			if (_model.has(id)) {
174 				if (prop) {
175 					setInnerHTML(node, Tools.getObjectsProperty(_model.get(id), prop));
176 				} else {
177 					setInnerHTML(node, _model.get(id));
178 				}
179 				
180 			}
181 			_observable.watch(id, function (newValue) {
182 				if (prop) {
183 					setInnerHTML(node, Tools.getObjectsProperty(_model.get(id), prop));
184 				} else {
185 					setInnerHTML(node, newValue);
186 				}
187 
188 			});
189 		};
190 	
191 		// Inits the model
192 		this.setModel($model);
193 		
194 		
195 	};
196 	
197 });
