  7 define("Olives/Model-plugin", 
  8 		
  9 ["Store", "Observable", "Tools", "Olives/DomUtils"],
 10 		
 11 /**
 12  * @class
 13  * This plugin links dom nodes to a model
 14  * @requires Store, Observable
 15  */
 16 function ModelPlugin(Store, Observable, Tools, DomUtils) {
 17 	
 18 	return function ModelPluginConstructor($model, $bindings) {
 19 		
 20 		/**
 21 		 * The model to watch
 22 		 * @private
 23 		 */
 24 		var _model = null,
 25 		
 26 		_bindings = {};
 27 		
 28 		/**
 29 		 * The observers handlers
 30 		 * for debugging only
 31 		 * @private
 32 		 */
 33 		this.observers = {};
 34 		
 35 		/**
 36 		 * Define the model to watch for
 37 		 * @param {Store} model the model to watch for changes
 38 		 * @returns {Boolean} true if the model was set
 39 		 */
 40 		this.setModel = function setModel(model) {
 41 			if (model instanceof Store) {
 42 				// Set the model
 43 				_model = model;
 44 				return true;
 45 			} else {
 46 				return false;
 47 			}
 48 		};
 49 		
 50 		/**
 51 		 * Get the store that is watched for
 52 		 * for debugging only
 53 		 * @private
 54 		 * @returns the Store
 55 		 */
 56 		this.getModel = function getModel() {
 57 			return _model;
 58 		};
 59 		
 60 		/**
 61 		 * The item renderer defines a dom node that can be duplicated
 62 		 * It is made available for debugging purpose, don't use it
 63 		 * @private
 64 		 */
 65 		this.ItemRenderer = function ItemRenderer(item) {
 66 			
 67 			var _node = null;
 68 			
 69 			/**
 70 			 * Set the duplicated node
 71 			 * @private
 72 			 */
 73 			this.set = function set(node) {
 74 				_node = node;
 75 				return true;
 76 			};
 77 			
 78 			/**
 79 			 * @private
 80 			 * @returns the node that is duplicated
 81 			 */
 82 			this.get = function get() {
 83 				return _node;
 84 			};
 85 			
 86 			/**
 87 			 * Associate the duplicated node to an item in the model
 88 			 * @private
 89 			 * @param id
 90 			 * @param pluginName
 91 			 * @returns the associated node
 92 			 */
 93 			this.associate = function associate(id, pluginName, hey) {
 94 				var newNode = _node.cloneNode(true);
 95 				var nodes = DomUtils.getNodes(newNode);
 96 
 97 				
 98 				Tools.toArray(nodes).forEach(function (child) {
 99             			child.dataset[pluginName+"_id"] = id;
100 				});
101 				return newNode;
102 			};
103 			
104 			this.set(item);			
105 		};
106 		
107 		/**
108 		 * Expands the inner dom nodes of a given dom node, filling it with model's values
109 		 * @param {HTMLElement} node the dom node to apply foreach to
110 		 */
111 		this.foreach = function foreach(node) {
112 			var domFragment,
113 				_rootNode = node.querySelector("*"),
114 				itemRenderer = new this.ItemRenderer(_rootNode);
115 
116 	
117 			domFragment = document.createDocumentFragment();
118             _model.loop(function (value, idx) {
119                 domFragment.appendChild(this.plugins.apply(itemRenderer.associate(idx, this.plugins.name)));
120             }, this);
121             
122 
123 	         node.replaceChild(domFragment, _rootNode);
124             
125             _model.watch("added", function (idx, value) {
126                 node.insertBefore(this.plugins.apply(itemRenderer.associate(idx, this.plugins.name)), node.childNodes[idx]);
127             }, this);
128             
129             _model.watch("deleted", function (idx) {
130                 node.removeChild(node.childNodes[idx]);
131                 this.observers[idx].forEach(function (handler) {
132                 	_model.unwatchValue(handler);
133                 }, this);
134                 delete this.observers[idx];
135             },this);
136          };
137 		
138 		/**
139 		 * Both ways binding between a dom node attributes and the model
140 		 * @param {HTMLElement} node the dom node to apply the plugin to
141 		 * @param {String} name the name of the property to look for in the model's value
142 		 * @returns
143 		 */
144 		this.bind = function bind(node, property, name) {
145 			
146 			// Name can be unset if the value of a row is plain text
147 			name = name || "";
148 
149 			// In case of an array-like model the id is the index of the model's item to look for.
150 			// The _id is added by the foreach function
151 			var id = node.dataset[this.plugins.name+"_id"],
152 		
153 			// Else, it is the first element of the following
154 			split = name.split("."),
155 			
156 			// So the index of the model is either id or the first element of split
157 			modelIdx = id || split.shift(),
158 			
159 			// And the name of the property to look for in the value is
160 			prop = id ? name : split.join("."),
161 			
162 			// Get the model's value
163 			get =  Tools.getNestedProperty(_model.get(modelIdx), prop);
164 			/**
165 			 * NEED SOME REFACTORING FROM HERE TO THE END OF THE FUNC
166 			 */
167 			// If truthy but 0 is a proper value too
168 			if (get || get === 0 || get === false) {
169 				if (_bindings[property]) {
170 					_bindings[property].call(node, get);
171 				} else {
172 					node[property] = get;
173 					
174 					// The test for the code placed here is not written
175 					// its : shouldn't double way data bind with plugins
176 					node.addEventListener("change", function (event) {
177 						if (prop) {
178 							var temp = _model.get(modelIdx);
179 							Tools.setNestedProperty(temp, name, node[property]);
180 							_model.set(modelIdx, temp);	
181 						} else {
182 							_model.set(modelIdx, node[property]);
183 						}
184 					}, true);
185 				}
186 			}
187 			
188 			// Watch for changes
189 			this.observers[modelIdx] = this.observers[modelIdx] || [];
190 			this.observers[modelIdx].push(_model.watchValue(modelIdx, function (value) {
191 				if (_bindings[property]) {
192 					_bindings[property].call(node, Tools.getNestedProperty(value, prop));
193 				} else {
194 					node[property] = Tools.getNestedProperty(value, prop);
195 				}
196 
197 			}));
198 		};
199 		
200 		/**
201 		 * Set the node's value into the model, the name is the model's property
202 		 * @param {HTMLElement} node
203 		 * @returns true if the property is added
204 		 */
205 		this.set = function set(node) {
206 			if (node instanceof HTMLElement && node.name) {
207 				_model.set(node.name, node.value);
208 				return true;
209 			} else {
210 				return false;
211 			}
212 		};
213 		
214 		/**
215 		 * Prevents the submit and set the model with all form's inputs
216 		 * @param {HTMLFormElement} form
217 		 * @returns true if valid form
218 		 */
219 		this.form = function form(form) {
220 			if (form && form.nodeName == "FORM") {
221 				var that = this;
222 				form.addEventListener("submit", function (event) {
223 					Tools.toArray(form.querySelectorAll("[name]")).forEach(that.set, that);
224 					event.preventDefault();
225 				}, true);
226 				return true;
227 			} else {
228 				return false;
229 			}
230 		};
231 		
232 		this.addBinding = function addBinding(name, binding) {
233 			if (name && typeof name == "string" && typeof binding == "function") {
234 				_bindings[name] = binding;
235 				return true;
236 			} else {
237 				return false;
238 			}
239 		};
240 		
241 		this.getBinding = function getBinding(name) {
242 			return _bindings[name];
243 		};
244 		
245 		this.addBindings = function addBindings(list) {
246 			return Tools.loop(list, function (binding, name) {
247 				this.addBinding(name, binding);
248 			}, this);
249 		};
250 	
251 		// Inits the model
252 		this.setModel($model);
253 		this.addBindings($bindings);
254 		
255 		
256 	};
257 	
258 });
