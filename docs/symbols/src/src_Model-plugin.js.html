1 define("Olives/Model-plugin", 
  2 		
  3 ["Store", "Observable"],
  4 		
  5 /**
  6  * @class
  7  * This plugin links dom nodes to a model
  8  * @requires Store, Observable
  9  */
 10 function ModelPlugin(Store, Observable) {
 11 	
 12 	return function ModelPluginConstructor($model) {
 13 		
 14 		/**
 15 		 * The model to watch
 16 		 * @private
 17 		 */
 18 		var _model = null,
 19 		
 20 		/**
 21 		 * An observable that apdapts to model's observable
 22 		 * @private
 23 		 */
 24 		_observable = new Observable(),
 25 		
 26 		/**
 27 		 * The function to put some content into a dom node
 28 		 * @private
 29 		 */
 30 		setInnerHTML = function setInnerHTML(dom, html) {
 31 			dom.innerHTML = html;
 32 		};
 33 		
 34 		/**
 35 		 * Define the model to watch for
 36 		 * @param {Store} model the model to watch for changes
 37 		 * @returns {Boolean} true if the model was set
 38 		 */
 39 		this.setModel = function setModel(model) {
 40 			if (model instanceof Store) {
 41 				// Set the model
 42 				_model = model;
 43 				// Also watch for value modification
 44 				_model.watch("updated", function (name, value) {
 45 					_observable.notify(name, value);
 46 				});
 47 				_model.watch("added", function (name, value) {
 48 					_observable.notify(name, value);
 49 				});
 50 				return true;
 51 			} else {
 52 				return false;
 53 			}
 54 		};
 55 		
 56 		/**
 57 		 * Get the store that is watched for
 58 		 * @returns the Store
 59 		 */
 60 		this.getModel = function getModel() {
 61 			return _model;
 62 		};
 63 		
 64 		/**
 65 		 * 
 66 		 * @returns
 67 		 */
 68 		this.toList = function toList() {
 69 			
 70 		};
 71 		
 72 		/**
 73 		 * Attach a model's value to a dom node so it also gets updated on value's changes
 74 		 * @param {HTMLElement} dom the dom node to apply the plugin to
 75 		 * @param {String} name the name of the model's value to attach it to
 76 		 * @returns
 77 		 */
 78 		this.toText = function toText(dom, name) {
 79 			// Leave the dom intact of no value
 80 			if (_model.has(name)) {
 81 				setInnerHTML(dom, _model.get(name));
 82 			}
 83 			// Watch for modifications
 84 			_observable.watch(name, function (value) {
 85 				setInnerHTML(dom, value);
 86 			});
 87 		};
 88 	
 89 		// Inits the model
 90 		this.setModel($model);
 91 		
 92 		
 93 	};
 94 	
 95 });
