  7 define("Olives/Model-plugin", 
  8 		
  9 ["Store", "Observable", "Tools", "Olives/DomUtils"],
 10 		
 11 /**
 12  * @class
 13  * This plugin links dom nodes to a model
 14  * @requires Store, Observable
 15  */
 16 function ModelPlugin(Store, Observable, Tools, DomUtils) {
 17 	
 18 	return function ModelPluginConstructor($model, $bindings) {
 19 		
 20 		/**
 21 		 * The model to watch
 22 		 * @private
 23 		 */
 24 		var _model = null,
 25 		
 26 		_bindings = {};
 27 		
 28 		/**
 29 		 * The observers handlers
 30 		 * for debugging only
 31 		 * @private
 32 		 */
 33 		this.observers = {};
 34 		
 35 		/**
 36 		 * Define the model to watch for
 37 		 * @param {Store} model the model to watch for changes
 38 		 * @returns {Boolean} true if the model was set
 39 		 */
 40 		this.setModel = function setModel(model) {
 41 			if (model instanceof Store) {
 42 				// Set the model
 43 				_model = model;
 44 				return true;
 45 			} else {
 46 				return false;
 47 			}
 48 		};
 49 		
 50 		/**
 51 		 * Get the store that is watched for
 52 		 * for debugging only
 53 		 * @private
 54 		 * @returns the Store
 55 		 */
 56 		this.getModel = function getModel() {
 57 			return _model;
 58 		};
 59 		
 60 		/**
 61 		 * The item renderer defines a dom node that can be duplicated
 62 		 * It is made available for debugging purpose, don't use it
 63 		 * @private
 64 		 */
 65 		this.ItemRenderer = function ItemRenderer($plugins, $rootNode) {
 66 			
 67 			/**
 68 			 * The node that will be cloned
 69 			 * @private
 70 			 */
 71 			var _node = null,
 72 			
 73 			/**
 74 			 * The object that contains plugins.name and plugins.apply
 75 			 * @private
 76 			 */
 77 			_plugins = null,
 78 			
 79 			/**
 80 			 * The _rootNode where to append the created items
 81 			 */
 82 			_rootNode = null;
 83 			
 84 			/**
 85 			 * Set the duplicated node
 86 			 * @private
 87 			 */
 88 			this.setRenderer = function setRenderer(node) {
 89 				_node = node;
 90 				return true;
 91 			};
 92 			
 93 			/**
 94 			 * @private
 95 			 * @returns the node that is duplicated
 96 			 */
 97 			this.getRenderer = function getRenderer() {
 98 				return _node;
 99 			};
100 			
101 			/**
102 			 * Sets the rootNode and gets the node to copy
103 			 * @param {HTMLElement} rootNode
104 			 * @returns
105 			 */
106 			this.setRootNode = function setRootNode(rootNode) {
107 				var renderer;
108 				if (rootNode instanceof HTMLElement) {
109 					_rootNode = rootNode;
110 					renderer = _rootNode.querySelector("*");
111 					this.setRenderer(renderer);
112 					_rootNode.removeChild(renderer);
113 					return true;
114 				} else {
115 					return false;
116 				}
117 			};
118 			
119 			/**
120 			 * Gets the rootNode
121 			 * @private
122 			 * @returns _rootNode
123 			 */
124 			this.getRootNode = function getRootNode() {
125 				return _rootNode;
126 			};
127 			
128 			/**
129 			 * Set the plugins objet that contains the name and the apply function
130 			 * @private
131 			 * @param plugins
132 			 * @returns true
133 			 */
134 			this.setPlugins = function setPlugins(plugins) {
135 				_plugins = plugins;
136 				return true;
137 			};
138 			
139 			/**
140 			 * Get the plugins object
141 			 * @private
142 			 * @returns the plugins object
143 			 */
144 			this.getPlugins = function getPlugins() {
145 				return _plugins;
146 			};
147 			
148 			/**
149 			 * The nodes created from the items are stored here
150 			 * @private
151 			 */
152 			this.items = new Store([]);
153 			
154 			/**
155 			 * Adds a new item and adds it in the items list
156 			 * @private
157 			 * @param {Number} id the id of the item
158 			 * @returns
159 			 */
160 			this.addItem = function addItem(id) {
161 				var node;
162 				if (typeof id == "number") {
163 					node = this.create(id);
164 					if (node) {
165 						_rootNode.insertBefore(this.create(id), this.items.get(id+1));
166 						return true;
167 					} else {
168 						return false;
169 					}
170 				} else {
171 					return false;
172 				}
173 			};
174 			
175 			/**
176 			 * Remove an item from the dom and the items list
177 			 * @private
178 			 * @param {Number} id the id of the item to remove
179 			 * @returns
180 			 */
181 			this.removeItem = function removeItem(id) {
182 				if (this.items.has(id)) {
183 					_rootNode.removeChild(this.items.get(id));
184 					this.items.del(id);
185 					return true;
186 				} else {
187 					return false;
188 				}
189 			};
190 			
191 			/**
192 			 * create a new node. Actually makes a clone of the initial one
193 			 * and adds pluginname_id to each node, then calls plugins.apply to apply all plugins
194 			 * @private
195 			 * @param id
196 			 * @param pluginName
197 			 * @returns the associated node
198 			 */
199 			this.create = function create(id) {
200 				if (_model.has(id)) {
201 					var newNode = _node.cloneNode(true),
202 					nodes = DomUtils.getNodes(newNode);
203 
204 					Tools.toArray(nodes).forEach(function (child) {
205 	            		child.dataset[_plugins.name+"_id"] = id;
206 					});
207 					
208 					this.items.set(id, newNode);
209 					_plugins.apply(newNode);
210 					return newNode;
211 				}
212 			};
213 			
214 			this.setPlugins($plugins);
215 			this.setRootNode($rootNode);
216 		};
217 		
218 		/**
219 		 * Expands the inner dom nodes of a given dom node, filling it with model's values
220 		 * @param {HTMLElement} node the dom node to apply foreach to
221 		 */
222 		this.foreach = function foreach(node, id, start, nb) {
223 			var itemRenderer = new this.ItemRenderer(this.plugins, node);
224 
225 			// Defines the boundaries
226 			start = start || 0;
227 			nb = nb || _model.getNbItems();
228 			var l = nb + start;
229 
230 			// Adds the items according to the boundaries
231 			for (var i=start; i<l; i++) {
232 				itemRenderer.addItem(i);
233 			}
234 
235 			// Add the newly created item
236             _model.watch("added", function (idx) {
237                 itemRenderer.addItem(idx);
238             }, this);
239             
240 			// If an item is deleted
241             _model.watch("deleted", function (idx) {
242                 itemRenderer.removeItem(idx);
243                 // Also remove all observers
244                 this.observers[idx].forEach(function (handler) {
245                 	_model.unwatchValue(handler);
246                 }, this);
247                 delete this.observers[idx];
248             },this);
249          };
250 		
251 		/**
252 		 * Both ways binding between a dom node attributes and the model
253 		 * @param {HTMLElement} node the dom node to apply the plugin to
254 		 * @param {String} name the name of the property to look for in the model's value
255 		 * @returns
256 		 */
257 		this.bind = function bind(node, property, name) {
258 
259 			// Name can be unset if the value of a row is plain text
260 			name = name || "";
261 
262 			// In case of an array-like model the id is the index of the model's item to look for.
263 			// The _id is added by the foreach function
264 			var id = node.dataset[this.plugins.name+"_id"],
265 		
266 			// Else, it is the first element of the following
267 			split = name.split("."),
268 			
269 			// So the index of the model is either id or the first element of split
270 			modelIdx = id || split.shift(),
271 			
272 			// And the name of the property to look for in the value is
273 			prop = id ? name : split.join("."),
274 			
275 			// Get the model's value
276 			get =  Tools.getNestedProperty(_model.get(modelIdx), prop);
277 
278 			// 0 and false are acceptable falsy values
279 			if (get || get === 0 || get === false) {
280 				// If the binding hasn't been overriden
281 				if (!this.execBinding(node, property, get)) {
282 					// Execute the default one which is a simple assignation
283 					node[property] = get;
284 				}
285 			}
286 			
287 			// Only watch for changes (double way data binding) if the binding
288 			// has not been redefined
289 			if (!this.hasBinding(property)) {
290 				node.addEventListener("change", function (event) {
291 					if (prop) {
292 						var temp = _model.get(modelIdx);
293 						Tools.setNestedProperty(temp, name, node[property]);
294 						_model.set(modelIdx, temp);	
295 					} else {
296 						_model.set(modelIdx, node[property]);
297 					}
298 				}, true);
299 
300 			}
301 			
302 			// Watch for changes
303 			this.observers[modelIdx] = this.observers[modelIdx] || [];
304 			this.observers[modelIdx].push(_model.watchValue(modelIdx, function (value) {
305 				if (!this.execBinding(node, property, Tools.getNestedProperty(value, prop))) {
306 					node[property] = Tools.getNestedProperty(value, prop);
307 				}
308 			}, this));
309 
310 		};
311 		
312 		/**
313 		 * Set the node's value into the model, the name is the model's property
314 		 * @private
315 		 * @param {HTMLElement} node
316 		 * @returns true if the property is added
317 		 */
318 		this.set = function set(node) {
319 			if (node instanceof HTMLElement && node.name) {
320 				_model.set(node.name, node.value);
321 				return true;
322 			} else {
323 				return false;
324 			}
325 		};
326 		
327 		/**
328 		 * Prevents the submit and set the model with all form's inputs
329 		 * @param {HTMLFormElement} form
330 		 * @returns true if valid form
331 		 */
332 		this.form = function form(form) {
333 			if (form && form.nodeName == "FORM") {
334 				var that = this;
335 				form.addEventListener("submit", function (event) {
336 					Tools.toArray(form.querySelectorAll("[name]")).forEach(that.set, that);
337 					event.preventDefault();
338 				}, true);
339 				return true;
340 			} else {
341 				return false;
342 			}
343 		};
344 		
345 		/**
346 		 * Add a new way to handle a binding
347 		 * @param {String} name of the binding
348 		 * @param {Function} binding the function to handle the binding
349 		 * @returns
350 		 */
351 		this.addBinding = function addBinding(name, binding) {
352 			if (name && typeof name == "string" && typeof binding == "function") {
353 				_bindings[name] = binding;
354 				return true;
355 			} else {
356 				return false;
357 			}
358 		};
359 		
360 		/**
361 		 * Execute a binding
362 		 * Only used by the plugin
363 		 * @private
364 		 * @param {HTMLElement} node the dom node on which to execute the binding
365 		 * @param {String} name the name of the binding
366 		 * @param {Any type} value the value to pass to the function
367 		 * @returns
368 		 */
369 		this.execBinding = function execBinding(node, name, value) {
370 			if (this.hasBinding(name)) {
371 				_bindings[name].call(node, value);
372 				return true;
373 			} else {
374 				return false;
375 			}
376 		};
377 		
378 		/**
379 		 * Check if the binding exists
380 		 * @private
381 		 * @param {String} name the name of the binding
382 		 * @returns
383 		 */
384 		this.hasBinding = function hasBinding(name) {
385 			return _bindings.hasOwnProperty(name);
386 		};
387 		
388 		/**
389 		 * Get a binding
390 		 * For debugging only
391 		 * @private
392 		 * @param {String} name the name of the binding
393 		 * @returns
394 		 */
395 		this.getBinding = function getBinding(name) {
396 			return _bindings[name];
397 		};
398 		
399 		/**
400 		 * Add multiple binding at once
401 		 * @param {Object} list the list of bindings to add
402 		 * @returns
403 		 */
404 		this.addBindings = function addBindings(list) {
405 			return Tools.loop(list, function (binding, name) {
406 				this.addBinding(name, binding);
407 			}, this);
408 		};
409 	
410 		// Inits the model
411 		this.setModel($model);
412 		// Inits bindings
413 		this.addBindings($bindings);
414 		
415 		
416 	};
417 	
418 });
