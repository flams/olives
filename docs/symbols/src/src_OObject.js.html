1 define("Olives/OObject", ["Store", "StateMachine", "Tools"],
  2 /** 
  3 * @class 
  4 * OObject is an abstract class that any UI can inherit from.
  5 * I will use UI for a piece of user interface, I've also could have called it a widget but UI's shorter.
  6 * It should provide code that is easy to reuse
  7 * @requires Store
  8 * @requires StateMachine
  9 * @requires Tools
 10 */
 11 function OObject(Store, StateMachine, Tools) {
 12 	
 13 	return function OObjectConstructor() {
 14 		
 15 		/**
 16 		 * This function creates the dom of the UI from it's template
 17 		 * It then queries the dom for data-connect to list them into this.connects
 18 		 * It can't be executed if the template is not set
 19 		 * @private
 20 		 */
 21 		var render = function render(args) {
 22 			var UI = args.UI;
 23 			
 24 			// If the template is set
 25 			if (UI.template) {
 26 				
 27 				// Here's the dom node that wraps the UI
 28 				UI.dom = _rootNode;
 29 				// In this function, the thisObject is the UI's prototype
 30 				// UI is the UI that has OObject as prototype
 31 				if (typeof UI.template == "string") {
 32 					// Let the browser do the parsing, can't be faster & easier.
 33 					UI.dom.innerHTML = UI.template;
 34 				} else if (UI.template instanceof HTMLElement) {
 35 					// If it's already an HTML element
 36 					UI.dom.appendChild(UI.template);
 37 				}
 38 				
 39 				// Populate the this.connects object with connect name : dom node
 40 				Tools.toArray(UI.dom.querySelectorAll("[data-connect]")).forEach(function (node) {
 41 					UI.connects[node.getAttribute("data-connect")] = node;
 42 				});
 43 				
 44 				// This function is empty and can be overridden by the user.
 45 				// It tells him that the UI is rendered
 46 				UI.onRender && UI.onRender();
 47 			} else {
 48 				// An explicit message I hope
 49 				throw Error("UI.template must be set prior to render");
 50 			}
 51 
 52 		},
 53 		
 54 		/**
 55 		 * This function appends the dom tree to the given dom node.
 56 		 * This dom node should be somewhere in the dom of the application
 57 		 * @private
 58 		 */
 59 		place = function place(args) {
 60 			var UI = args.UI;
 61 			// Append child appends but can also detach the dom from its previous position
 62 			args.params && args.params.appendChild(UI.dom);
 63 			UI.onPlace && UI.onPlace();
 64 		},
 65 		
 66 		/**
 67 		 * Does rendering & placing in one function
 68 		 * @private
 69 		 */
 70 		renderNPlace = function renderNPlace(args) {
 71 			render(args);
 72 			place(args);
 73 		},
 74 
 75 		/**
 76 		 * This is the UI's wrapper.
 77 		 * @private HTMLElement
 78 		 */
 79 		_rootNode = document.createElement("div"),
 80 			
 81 		/**
 82 		 * The UI's stateMachine.
 83 		 * Much better than if(stuff) do(stuff) else if (!stuff and stuff but not stouff) do (otherstuff)
 84 		 * Please open an issue if you want to propose a better one
 85 		 * @private
 86 		 */
 87 		_stateMachine = new StateMachine("Init", {
 88 			"Init": [["render", render, this, "Rendered"],
 89 			         ["place", renderNPlace, this, "Rendered"]],
 90 			"Rendered": [["place", place, this],
 91 			             ["render", render, this]]
 92 		});
 93 		
 94 		/**
 95 		 * The UI's Store
 96 		 * It has set/get/del/has/watch/unwatch methods
 97 		 * @see Emily's doc for more info on how it works.
 98 		 */
 99 		this.model = new Store();
100 		
101 		/**
102 		 * Describes the template, can either be like "<p></p>" or HTMLElements
103 		 * @type string or HTMLElement
104 		 */
105 		this.template = null;
106 		
107 		/**
108 		 * This will hold the dom nodes built from the template.
109 		 */
110 		this.dom = null;
111 		
112 		/**
113 		 * Connects associates dom nodes with their data-connect attribute
114 		 */
115 		this.connects = {};
116 		
117 		/**
118 		 * This function can be overridden. 
119 		 * It's called when the rendering is done.
120 		 */
121 		this.onRender = function onRender() {
122 			
123 		};
124 			
125 		/**
126 		 * Action triggers a state change, such as "place" or "render.
127 		 * Watch the life cycle documentation for more info.
128 		 * @param {String} name the name of the action
129 		 * @param {Object} params the parameters to pass to the actions
130 		 */
131 		this.action = function action(name, params) {
132 			_stateMachine.event(name, {UI: this,
133 				params: params});
134 		};
135 		
136 		/**
137 		 * This function can be overridden. 
138 		 * It's called when the placing is done.
139 		 */
140 		this.onPlace = function onPlace() {
141 			
142 		};
143 		
144 	};
145 	
146 });
