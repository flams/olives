  7 define("Olives/OObject", ["StateMachine", "Store", "Olives/Plugins", "Olives/DomUtils", "Tools"],
  8 /** 
  9 * @class 
 10 * OObject is an abstract class that any UI can inherit from.
 11 * It should provide code that is easy to reuse
 12 * @requires StateMachine
 13 */
 14 function OObject(StateMachine, Store, Plugins, DomUtils, Tools) {
 15 	
 16 	return function OObjectConstructor(otherStore) {
 17 		
 18 		/**
 19 		 * This function creates the dom of the UI from it's template
 20 		 * It then queries the dom for data-model to list them into this.connects
 21 		 * It can't be executed if the template is not set
 22 		 * @private
 23 		 */
 24 		var render = function render(UI) {
 25 			
 26 			// The place where the template will be created
 27 			// is either the currentPlace where the node is placed
 28 			// or a temporary div
 29 			var baseNode = _currentPlace || document.createElement("div");
 30 			
 31 			// If the template is set
 32 			if (UI.template) {
 33 				// In this function, the thisObject is the UI's prototype
 34 				// UI is the UI that has OObject as prototype
 35 				if (typeof UI.template == "string") {
 36 					// Let the browser do the parsing, can't be faster & easier.
 37 					baseNode.innerHTML = UI.template.trim();
 38 				} else if (UI.template instanceof HTMLElement) {
 39 					// If it's already an HTML element
 40 					baseNode.appendChild(UI.template);
 41 				}
 42 				
 43 				// The UI must be placed in a unique dom node
 44 				// If not, there can't be multiple UIs placed in the same parentNode
 45 				// as it wouldn't be possible to know which node would belong to which UI
 46 				// This is probably a DOM limitation.
 47 				if (baseNode.childNodes.length > 1) {
 48 					throw Error("UI.template should have only one parent node");
 49 				} else {
 50 					UI.dom = baseNode.childNodes[0];
 51 				}
 52 				
 53 				UI.alive(UI.dom);
 54 
 55 			} else {
 56 				// An explicit message I hope
 57 				throw Error("UI.template must be set prior to render");
 58 			}
 59 		},
 60 		
 61 		/**
 62 		 * This function appends the dom tree to the given dom node.
 63 		 * This dom node should be somewhere in the dom of the application
 64 		 * @private
 65 		 */
 66 		place = function place(UI, place) {
 67 			if (place) {
 68 				place.appendChild(UI.dom);
 69 				// Also save the new place, so next renderings
 70 				// will be made inside it
 71 				_currentPlace = place;
 72 			}
 73 		},
 74 		
 75 		/**
 76 		 * Does rendering & placing in one function
 77 		 * @private
 78 		 */
 79 		renderNPlace = function renderNPlace(UI, dom) {
 80 			render(UI);
 81 			place(UI, dom);
 82 		},
 83 		
 84 		/**
 85 		 * This stores the current place
 86 		 * If this is set, this is the place where new templates
 87 		 * will be appended
 88 		 * @private
 89 		 */
 90 		_currentPlace = null, 
 91 		
 92 		/**
 93 		 * The UI's stateMachine.
 94 		 * Much better than if(stuff) do(stuff) else if (!stuff and stuff but not stouff) do (otherstuff)
 95 		 * Please open an issue if you want to propose a better one
 96 		 * @private
 97 		 */
 98 		_stateMachine = new StateMachine("Init", {
 99 			"Init": [["render", render, this, "Rendered"],
100 			         ["place", renderNPlace, this, "Rendered"]],
101 			"Rendered": [["place", place, this],
102 			             ["render", render, this]]
103 		});
104 		
105 		/**
106 		 * The UI's Store
107 		 * It has set/get/del/has/watch/unwatch methods
108 		 * @see Emily's doc for more info on how it works.
109 		 */
110 		this.model = otherStore instanceof Store ? otherStore : new Store;
111 		
112 		/**
113 		 * The module that will manage the plugins for this UI
114 		 * @see Olives/Plugins' doc for more info on how it works.
115 		 */
116 		this.plugins = new Plugins();
117 		
118 		/**
119 		 * Describes the template, can either be like "<p></p>" or HTMLElements
120 		 * @type string or HTMLElement
121 		 */
122 		this.template = null;
123 		
124 		/**
125 		 * This will hold the dom nodes built from the template.
126 		 */
127 		this.dom = null;
128 		
129 		/**
130 		 * Place the UI in a given dom node
131 		 * @param {HTMLElement} node the node on which to append the UI
132 		 */
133 		this.place = function place(node) {
134 			_stateMachine.event("place", this, node);
135 		};
136 		
137 		/**
138 		 * Renders the template to dom nodes and applies the plugins on it
139 		 * It requires the template to be set first
140 		 */
141 		this.render = function render() {
142 			_stateMachine.event("render", this);
143 		};
144 		
145 		/**
146 		 * Applies this UI's plugins to the given dom node,
147 		 * kind of making it 'alive'
148 		 * @param {HTMLElement} node the dom to apply the plugins to
149 		 * @returns false if wrong param
150 		 */
151 		this.alive = function alive(dom) {
152 			if (dom instanceof HTMLElement) {
153 				this.plugins.apply(dom);
154 				return true;
155 			} else {
156 				return false;
157 			}
158 			
159 		};
160 		
161 	};
162 	
163 });
