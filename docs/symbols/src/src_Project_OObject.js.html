1 define("Olives/OObject", ["TinyStore", "StateMachine"],
  2 /** 
  3 * @class 
  4 * OObject is an abstract class that any UI can inherit from.
  5 * It should provide code that is easy to reuse
  6 */
  7 function OObject(TinyStore, StateMachine) {
  8 	
  9 	function OObject() {
 10 		
 11 		var render = function render(args) {
 12 				var UI = args.UI;
 13 				
 14 				if (UI.template) {
 15 					UI.dom = _rootNode;
 16 					// In this function, the thisObject is the UI's prototype
 17 					// UI is the UI that has OObject as prototype
 18 					if (typeof UI.template == "string") {
 19 						UI.dom.innerHTML = UI.template;
 20 					} else if (UI.template instanceof HTMLElement) {
 21 						UI.dom.appendChild(UI.template);
 22 					}
 23 					UI.onRender && UI.onRender();
 24 				} else {
 25 					throw "UI.template must be set prior to render";
 26 				}
 27 
 28 			},
 29 			place = function place(args) {
 30 				var UI = args.UI;
 31 				args.params && args.params.appendChild(UI.dom);
 32 				UI.onPlace && UI.onPlace();
 33 			},
 34 			renderNPlace = function renderNPlace(args) {
 35 				render(args);
 36 				place(args);
 37 			},
 38 		
 39 		_template = null,
 40 		
 41 		_rootNode = document.createElement("div"),
 42 			
 43 		_stateMachine = StateMachine.create("Init", {
 44 			"Init": [["render", render, this, "Rendered"],
 45 			         ["place", renderNPlace, this]],
 46 			"Rendered": [["place", place, this],
 47 			             ["render", render, this]]
 48 		});
 49 		
 50 		this.model = TinyStore.create();
 51 		
 52 		this.template = null;
 53 		
 54 		this.dom = null;
 55 		
 56 		this.onRender = function onRender() {
 57 			
 58 		};
 59 			
 60 		this.action = function action(name, params) {
 61 			_stateMachine.event(name, {UI: this,
 62 				params: params});
 63 		};
 64 		
 65 		this.onPlace = function onPlace() {
 66 			
 67 		};
 68 		
 69 	};
 70 	
 71 	return {
 72 		augment: function augment(func) {
 73 			func.prototype = new OObject;
 74 			return func;
 75 		},
 76 		isAugmenting: function isAugmenting(module) {
 77 			var proto = module.prototype ? module.prototype : Object.getPrototypeOf(module);
 78 			return proto instanceof OObject;
 79 		}
 80 	};
 81 	
 82 });
