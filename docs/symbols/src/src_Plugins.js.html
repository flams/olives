1 define("Olives/Plugins", ["Tools", "Olives/DomUtils"],
  2 		
  3 /**
  4  * @class
  5  * Plugins is the link between the UI and your plugins.
  6  * You can design your own plugin, declare them in your UI, and call them
  7  * from the template, like :
  8  * <tag data-yourPlugin="method: param"></tag>
  9  * @see Model-Plugin for instance
 10  * @requires Tools
 11  */
 12 function Plugins(Tools, DomUtils) {
 13 	
 14 	return function PluginsConstructor() {
 15 		
 16 		/**
 17 		 * The list of plugins
 18 		 * @private
 19 		 */
 20 		var _plugins = {},
 21 
 22 		/**
 23 		 * Just a "functionalification" of trim
 24 		 * for code readability
 25 		 * @private
 26 		 */
 27 		trim = function trim(string) {
 28 			return string.trim();
 29 		},
 30 			
 31 		/**
 32 		 * Call the plugins methods, passing them the dom node
 33 		 * A phrase can be :
 34 		 * <tag data-plugin='method: param, param; method:param...'/>
 35 		 * the function has to call every method of the plugin
 36 		 * passing it the node, and the given params
 37 		 * @private
 38 		 */
 39 		applyPlugin = function applyPlugin(node, phrase, plugin) {
 40 			// Split the methods
 41 			phrase.split(";")
 42 			.forEach(function (couple) {
 43 				// Split the result between method and params
 44 				var split = couple.split(":"),
 45 				// Trim the name
 46 				method = split[0].trim(),
 47 				// And the params, if any
 48 				params = split[1] ? split[1].split(",").map(trim) : [];
 49 				
 50 				// The first param must be the dom node
 51 				params.unshift(node);
 52 
 53 				if (_plugins[plugin] && _plugins[plugin][method]) {
 54 					// Call the method with the following params for instance :
 55 					// [node, "param1", "param2" .. ]
 56 					_plugins[plugin][method].apply(_plugins[plugin], params);
 57 				}
 58 
 59 			});
 60 		};
 61 		
 62 		/**
 63 		 * Add a plugin
 64 		 * 
 65 		 * Note that once added, the function adds a "plugins" property to the plugin.
 66 		 * It's an object that holds a name property, with the registered name of the plugin
 67 		 * and an apply function, to use on new nodes that the plugin would generate
 68 		 * 
 69 		 * @param {String} name the name of the data that the plugin should look for
 70 		 * @param {Object} plugin the plugin that has the functions to execute
 71 		 * @returns true if plugin successfully added.
 72 		 */
 73 		this.add = function add(name, plugin, override) {
 74 			var that = this,
 75 				propertyName = "plugins";
 76 			
 77 			if (typeof name == "string" && typeof plugin == "object" && plugin) {
 78 				_plugins[name] = plugin;
 79 				
 80 				plugin[propertyName] = {
 81 						name: name,
 82 						apply: function apply() {
 83 							return that.apply.apply(that, arguments);
 84 						}
 85 				};
 86 				return true;
 87 			} else {
 88 				return false;
 89 			}
 90 		};
 91 		
 92 		/**
 93 		 * Get a previously added plugin
 94 		 * @param {String} name the name of the plugin
 95 		 * @returns {Object} the plugin
 96 		 */
 97 		this.get = function get(name) {
 98 			return _plugins[name];
 99 		};
100 		
101 		/**
102 		 * Delete a plugin from the list
103 		 * @param {String} name the name of the plugin
104 		 * @returns {Boolean} true if success
105 		 */
106 		this.del = function del(name) {
107 			return delete _plugins[name];
108 		};
109 		
110 		/**
111 		 * Apply the plugins to a NodeList
112 		 * @param {HTMLElement} dom the dom nodes on which to apply the plugins
113 		 * @returns {Boolean} true if the param is a dom node
114 		 */
115 		this.apply = function apply(dom) {
116 			
117 			var nodes;
118 			
119 			if (dom instanceof HTMLElement) {
120 				
121 				nodes = DomUtils.getNodes(dom);
122 				Tools.loop(Tools.toArray(nodes), function (node) {
123 					Tools.loop(node.dataset, function (phrase, plugin) {
124 						applyPlugin(node, phrase, plugin);
125 					});
126 				});
127 				
128 				return dom;
129 				
130 			} else {
131 				return false;
132 			}
133 		};
134 		
135 	};
136 });
