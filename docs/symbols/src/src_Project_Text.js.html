1 define("Olives/Text",
  2 	
  3 ["Olives/OObject"],
  4 
  5 /** 
  6 * @class 
  7 * Text is a simple paragraph with a unique property that shows the way UI work.
  8 */		
  9 function Text(OObject) {
 10 	
 11 	function _Text(content) {
 12 		
 13 		/**
 14 		 * The UI's template that will generate the dom nodes to be placed on the application's page.
 15 		 */
 16 		this.template = "<p data-connect='content'></p>";
 17 		
 18 		/**
 19 		 * Set the paragraph's innherHTML with given content
 20 		 * @param {String} content the text
 21 		 */
 22 		this.setContent = function setContent(content) {
 23 			this.connects["content"].innerHTML = content;
 24 		};
 25 
 26 		/**
 27 		 * onRender is called after action("render")
 28 		 */
 29 		this.onRender = function onRender() {
 30 			this.setContent(this.model.get("content"));
 31 		};
 32 
 33 		// If a content is given at init, set it.
 34 		content && this.model.set("content", content);
 35 		// Watch for its modifications
 36 		this.model.watch("content", this.setContent, this);	
 37 		
 38 	}
 39 	
 40 	return {
 41 		/**
 42 		 * Create a new Text UI
 43 		 * @returns the new ui
 44 		 */
 45 		create: function create(content) {
 46 			var augmentedText = OObject.augment(_Text);
 47 			var text = new augmentedText(content);
 48 			return text;
 49 		}
 50 	};
 51 
 52 	
 53 });
