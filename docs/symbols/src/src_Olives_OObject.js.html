1 define("Olives/OObject", ["Tools", "TinyStore"],
  2 /** 
  3 * @class 
  4 * _base is an abstract class that every UI can inherit from.
  5 * It should provide code that is easy to reuse
  6 */
  7 function OObject(Tools, Tinystore) {
  8 	
  9 	return {
 10 		/**
 11 		 * The model of the UI is a TinyStore
 12 		 * @type object
 13 		 */
 14 		model: Tinystore.create(),
 15 	
 16 		/**
 17 		 * The UI's template.
 18 		 * @type HTMLElement
 19 		 */
 20 		template: null,
 21 		
 22 		/**
 23 		 * Connects contains references to dom nodes with data-connect attributes
 24 		 * @type object 
 25 		 */
 26 		connects: {},
 27 		
 28 		/**
 29 		 * Render the UI at a given place (dom node)
 30 		 * @param {HTMLElement} rootNode the node where to append the UI
 31 		 * @returns {Object} the UI
 32 		 */
 33 		place: function place(parentNode) {
 34 			
 35 			Tools.toArray(this.template.querySelectorAll("[data-connect]")).forEach(function (node) {
 36 				this.connects[node.getAttribute("data-connect")] = node;
 37 			}, this);
 38 	
 39 			parentNode.appendChild(this.template);		
 40 			return parentNode;
 41 		}
 42 	};
 43 	
 44 });
