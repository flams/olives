  7 define(
  8 
  9 /**
 10  * @class
 11  * A Stack is a tool for managing DOM elements as groups. Within a group, dom elements
 12  * can be added, removed, moved around. The group can be moved to another parent node
 13  * while keeping the DOM elements in the same order, excluding the parent dom elements's
 14  * children that are not in the Stack.
 15  */
 16 function Stack() {
 17 
 18 	"use strict";
 19 
 20 	var Tools = require("Tools");
 21 
 22 	return function StackConstructor($parent) {
 23 
 24 		/**
 25 		 * The parent DOM element is a documentFragment by default
 26 		 * @private
 27 		 */
 28 		var _parent = document.createDocumentFragment(),
 29 
 30 		/**
 31 		 * The place where the dom elements hide
 32 		 * @private
 33 		 */
 34 		_hidePlace = document.createElement("div"),
 35 
 36 		/**
 37 		 * The list of dom elements that are part of the stack
 38 		 * Helps for excluding elements that are not part of it
 39 		 * @private
 40 		 */
 41 		_childNodes = [],
 42 
 43 		_lastTransit = null;
 44 
 45 		/**
 46 		 * Add a DOM element to the stack. It will be appended.
 47 		 * @param {HTMLElement} dom the DOM element to add
 48 		 * @returns {HTMLElement} dom
 49 		 */
 50 		this.add = function add(dom) {
 51 			if (!this.has(dom) && dom instanceof HTMLElement) {
 52 				_parent.appendChild(dom);
 53 				_childNodes.push(dom);
 54 				return dom;
 55 			} else {
 56 				return false;
 57 			}
 58 		};
 59 
 60 		/**
 61 		 * Remove a DOM element from the stack.
 62 		 * @param {HTMLElement} dom the DOM element to remove
 63 		 * @returns {HTMLElement} dom
 64 		 */
 65 		this.remove = function remove(dom) {
 66 			var index;
 67 			if (this.has(dom)) {
 68 				index = _childNodes.indexOf(dom);
 69 				_parent.removeChild(dom);
 70 				_childNodes.splice(index, 1);
 71 				return dom;
 72 			} else {
 73 				return false;
 74 			}
 75 		};
 76 
 77 		/**
 78 		 * Place a stack by appending its DOM elements to a new parent
 79 		 * @param {HTMLElement} newParentDom the new DOM element to append the stack to
 80 		 * @returns {HTMLElement} newParentDom
 81 		 */
 82 		this.place = function place(newParentDom) {
 83 			if (newParentDom instanceof HTMLElement) {
 84 				[].slice.call(_parent.childNodes).forEach(function (childDom) {
 85 					if (this.has(childDom)) {
 86 						newParentDom.appendChild(childDom);
 87 					}
 88 				}, this);
 89 				return this._setParent(newParentDom);
 90 			} else {
 91 				return false;
 92 			}
 93 		};
 94 
 95 		/**
 96 		 * Move an element up in the stack
 97 		 * @param {HTMLElement} dom the dom element to move up
 98 		 * @returns {HTMLElement} dom
 99 		 */
100 		this.up = function up(dom) {
101 			if (this.has(dom)) {
102 				var domPosition = this.getPosition(dom);
103 				this.move(dom, domPosition + 1);
104 				return dom;
105 			} else {
106 				return false;
107 			}
108 		};
109 
110 		/**
111 		 * Move an element down in the stack
112 		 * @param {HTMLElement} dom the dom element to move down
113 		 * @returns {HTMLElement} dom
114 		 */
115 		this.down = function down(dom) {
116 			if (this.has(dom)) {
117 				var domPosition = this.getPosition(dom);
118 				this.move(dom, domPosition - 1);
119 				return dom;
120 			} else {
121 				return false;
122 			}
123 		};
124 
125 		/**
126 		 * Move an element that is already in the stack to a new position
127 		 * @param {HTMLElement} dom the dom element to move
128 		 * @param {Number} position the position to which to move the DOM element
129 		 * @returns {HTMLElement} dom
130 		 */
131 		this.move = function move(dom, position) {
132 			if (this.has(dom)) {
133 				var domIndex = _childNodes.indexOf(dom);
134 				_childNodes.splice(domIndex, 1);
135 				// Preventing a bug in IE when insertBefore is not given a valid
136 				// second argument
137 				var nextElement = getNextElementInDom(position);
138 				if (nextElement) {
139 					_parent.insertBefore(dom, nextElement);
140 				} else {
141 					_parent.appendChild(dom);
142 				}
143 				_childNodes.splice(position, 0, dom);
144 				return dom;
145 			} else {
146 				return false;
147 			}
148 		};
149 
150 		function getNextElementInDom(position) {
151 			if (position >= _childNodes.length) {
152 				return;
153 			}
154 			var nextElement = _childNodes[position];
155 			if (Tools.toArray(_parent.childNodes).indexOf(nextElement) == -1) {
156 				return getNextElementInDom(position +1);
157 			} else {
158 				return nextElement;
159 			}
160 		}
161 
162 		/**
163 		 * Insert a new element at a specific position in the stack
164 		 * @param {HTMLElement} dom the dom element to insert
165 		 * @param {Number} position the position to which to insert the DOM element
166 		 * @returns {HTMLElement} dom
167 		 */
168 		this.insert = function insert(dom, position) {
169 			if (!this.has(dom) && dom instanceof HTMLElement) {
170 				_childNodes.splice(position, 0, dom);
171 				_parent.insertBefore(dom, _parent.childNodes[position]);
172 				return dom;
173 			} else {
174 				return false;
175 			}
176 		};
177 
178 		/**
179 		 * Get the position of an element in the stack
180 		 * @param {HTMLElement} dom the dom to get the position from
181 		 * @returns {HTMLElement} dom
182 		 */
183 		this.getPosition = function getPosition(dom) {
184 			return _childNodes.indexOf(dom);
185 		};
186 
187 		/**
188 		 * Count the number of elements in a stack
189 		 * @returns {Number} the number of items
190 		 */
191 		this.count = function count() {
192 			return _parent.childNodes.length;
193 		};
194 
195 		/**
196 		 * Tells if a DOM element is in the stack
197 		 * @param {HTMLElement} dom the dom to tell if its in the stack
198 		 * @returns {HTMLElement} dom
199 		 */
200 		this.has = function has(childDom) {
201 			return this.getPosition(childDom) >= 0;
202 		};
203 
204 		/**
205 		 * Hide a dom element that was previously added to the stack
206 		 * It will be taken out of the dom until displayed again
207 		 * @param {HTMLElement} dom the dom to hide
208 		 * @return {boolean} if dom element is in the stack
209 		 */
210 		this.hide = function hide(dom) {
211 			if (this.has(dom)) {
212 				_hidePlace.appendChild(dom);
213 				return true;
214 			} else {
215 				return false;
216 			}
217 		};
218 
219 		/**
220 		 * Show a dom element that was previously hidden
221 		 * It will be added back to the dom
222 		 * @param {HTMLElement} dom the dom to show
223 		 * @return {boolean} if dom element is current hidden
224 		 */
225 		this.show = function show(dom) {
226 			if (this.has(dom) && dom.parentNode === _hidePlace) {
227 				this.move(dom, _childNodes.indexOf(dom));
228 				return true;
229 			} else {
230 				return false;
231 			}
232 		};
233 
234 		/**
235 		 * Helper function for hiding all the dom elements
236 		 */
237 		this.hideAll = function hideAll() {
238 			_childNodes.forEach(this.hide, this);
239 		};
240 
241 		/**
242 		 * Helper function for showing all the dom elements
243 		 */
244 		this.showAll = function showAll() {
245 			_childNodes.forEach(this.show, this);
246 		};
247 
248 		/**
249 		 * Get the parent node that a stack is currently attached to
250 		 * @returns {HTMLElement} parent node
251 		 */
252 		this.getParent = function _getParent() {
253 				return _parent;
254 		};
255 
256 		/**
257 		 * Set the parent element (without appending the stacks dom elements to)
258 		 * @private
259 		 */
260 		this._setParent = function _setParent(parent) {
261 			if (parent instanceof HTMLElement) {
262 				_parent = parent;
263 				return _parent;
264 			} else {
265 				return false;
266 			}
267 		};
268 
269 		/**
270 		 * Get the place where the DOM elements are hidden
271 		 * @private
272 		 */
273 		this.getHidePlace = function getHidePlace() {
274 			return _hidePlace;
275 		};
276 
277 		/**
278 		 * Set the place where the DOM elements are hidden
279 		 * @private
280 		 */
281 		this.setHidePlace = function setHidePlace(hidePlace) {
282 			if (hidePlace instanceof HTMLElement) {
283 				_hidePlace = hidePlace;
284 				return true;
285 			} else {
286 				return false;
287 			}
288 		};
289 
290 		/**
291 		 * Get the last dom element that the stack transitted to
292 		 * @returns {HTMLElement} the last dom element
293 		 */
294 		this.getLastTransit = function getLastTransit() {
295 			return _lastTransit;
296 		};
297 
298 		/**
299 		 * Transit between views, will show the new one and hide the previous
300 		 * element that the stack transitted to, if any.
301 		 * @param {HTMLElement} dom the element to transit to
302 		 * @returns {Boolean} false if the element can't be shown
303 		 */
304 		this.transit = function transit(dom) {
305 			if (_lastTransit) {
306 				this.hide(_lastTransit);
307 			}
308 			if (this.show(dom)) {
309 				_lastTransit = dom;
310 				return true;
311 			} else {
312 				return false;
313 			}
314 		};
315 
316 		this._setParent($parent);
317 
318 	};
319 
320 });
