  7 define(["Router", "Tools"],
  8 
  9 /**
 10  * @class
 11  * A locationRouter is a router which navigates to the route defined in the URL and updates this URL
 12  * while navigating. It's a subtype of Emily's Router
 13  */
 14 function LocationRouter(Router, Tools) {
 15 
 16     "use strict";
 17 
 18     function LocationRouterConstructor() {
 19 
 20         /**
 21          * The handle on the watch
 22          * @private
 23          */
 24         var _watchHandle,
 25 
 26         /**
 27          * The default route to navigate to when nothing is supplied in the url
 28          * @private
 29          */
 30         _defaultRoute = "",
 31 
 32         /**
 33          * The last route that was navigated to
 34          * @private
 35          */
 36         _lastRoute = window.location.hash;
 37 
 38         /**
 39          * Navigates to the current hash or to the default route if none is supplied in the url
 40          * @private
 41          */
 42          /*jshint validthis:true*/
 43         function doNavigate() {
 44             if (window.location.hash) {
 45                 var parsedHash = this.parse(window.location.hash);
 46                 this.navigate.apply(this, parsedHash);
 47             } else {
 48                 this.navigate(_defaultRoute);
 49             }
 50         }
 51 
 52         /**
 53          * Set the default route to navigate to when nothing is defined in the url
 54          * @param {String} defaultRoute the defaultRoute to navigate to
 55          * @returns {Boolean} true if it's not an empty string
 56          */
 57         this.setDefaultRoute = function setDefaultRoute(defaultRoute) {
 58             if (defaultRoute && typeof defaultRoute == "string") {
 59                 _defaultRoute = defaultRoute;
 60                 return true;
 61             } else {
 62                 return false;
 63             }
 64         };
 65 
 66         /**
 67          * Get the currently set default route
 68          * @returns {String} the default route
 69          */
 70         this.getDefaultRoute = function getDefaultRoute() {
 71             return _defaultRoute;
 72         };
 73 
 74         /**
 75          * The function that parses the url to determine the route to navigate to.
 76          * It has a default behavior explained below, but can be overriden as long as
 77          * it has the same contract.
 78          * @param {String} hash the hash coming from window.location.has
 79          * @returns {Array} has to return an array with the list of arguments to call
 80          *    navigate with. The first item of the array must be the name of the route.
 81          *
 82          * Example: #album/holiday/2013
 83          *      will navigate to the route "album" and give two arguments "holiday" and "2013"
 84          */
 85         this.parse = function parse(hash) {
 86             return hash.split("#").pop().split("/");
 87         };
 88 
 89         /**
 90          * The function that converts, or serialises the route and its arguments to a valid URL.
 91          * It has a default behavior below, but can be overriden as long as it has the same contract.
 92          * @param {Array} args the list of arguments to serialize
 93          * @returns {String} the serialized arguments to add to the url hashmark
 94          *
 95          * Example:
 96          *      ["album", "holiday", "2013"];
 97          *      will give "album/holiday/2013"
 98          *
 99          */
100         this.toUrl = function toUrl(args) {
101             return args.join("/");
102         };
103 
104         /**
105          * When all the routes and handlers have been defined, start the location router
106          * so it parses the URL and navigates to the corresponding route.
107          * It will also start listening to route changes and hashmark changes to navigate.
108          * While navigating, the hashmark itself will also change to reflect the current route state
109          */
110         this.start = function start(defaultRoute) {
111             this.setDefaultRoute(defaultRoute);
112             doNavigate.call(this);
113             this.bindOnHashChange();
114             this.bindOnRouteChange();
115         };
116 
117         /**
118          * Remove the events handler for cleaning.
119          */
120         this.destroy = function destroy() {
121             this.unwatch(_watchHandle);
122             window.removeEventListener("hashchange", this.boundOnHashChange, true);
123         };
124 
125         /**
126          * Parse the hash and navigate to the corresponding url
127          * @private
128          */
129         this.onHashChange  = function onHashChange() {
130             if (window.location.hash != _lastRoute) {
131                 doNavigate.call(this);
132             }
133         };
134 
135         /**
136          * The bound version of onHashChange for add/removeEventListener
137          * @private
138          */
139         this.boundOnHashChange = this.onHashChange.bind(this);
140 
141         /**
142          * Add an event listener to hashchange to navigate to the corresponding route
143          * when it changes
144          * @private
145          */
146         this.bindOnHashChange = function bindOnHashChange() {
147             window.addEventListener("hashchange", this.boundOnHashChange, true);
148         };
149 
150         /**
151          * Watch route change events from the router to update the location
152          * @private
153          */
154         this.bindOnRouteChange = function bindOnRouteChange() {
155             _watchHandle = this.watch(this.onRouteChange, this);
156         };
157 
158         /**
159          * The handler for when the route changes
160          * It updates the location
161          * @private
162          */
163         this.onRouteChange = function onRouteChange() {
164             window.location.hash = this.toUrl(Tools.toArray(arguments));
165             _lastRoute = window.location.hash;
166         };
167 
168         this.getLastRoute = function getLastRoute() {
169             return _lastRoute;
170         };
171 
172     }
173 
174     return function LocationRouterFactory() {
175         LocationRouterConstructor.prototype = new Router();
176         return new LocationRouterConstructor();
177     };
178 
179 });
