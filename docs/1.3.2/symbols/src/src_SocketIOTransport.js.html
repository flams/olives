  7 define(["Observable", "Tools"],
  8 /**
  9  * @class
 10  * Transport allows for client-server eventing.
 11  * It's based on socket.io.
 12  */
 13 function Transport(Observable, Tools) {
 14 
 15 	/**
 16 	 * Defines the Transport
 17 	 * @private
 18 	 * @param {Object} $io socket.io's object
 19 	 * @param {url} $url the url to connect Transport to
 20 	 * @returns
 21 	 */
 22 	return function TransportConstructor($io, $url) {
 23 
 24 		/**
 25 		 * @private
 26 		 * The socket.io's socket
 27 		 */
 28 		var _socket = null,
 29 
 30 		/**
 31 		 * @private
 32 		 * The socket.io globally defined module
 33 		 */
 34 		_io = null,
 35 
 36 		/**
 37 		 * @private
 38 		 * The Observable that is used for the listen function
 39 		 */
 40 		_observable = new Observable(),
 41 
 42 		/**
 43 		 * @private
 44 		 * listen internally calls request, which returns a "stop listen" function
 45 		 * It should be saved in an object alongside the topic name
 46 		 */
 47 		_stops = {};
 48 
 49 		/**
 50 		 * Set the io handler (socket.io)
 51 		 * @param {Object} io the socket.io object
 52 		 * @returns true if it seems to be socket.io
 53 		 */
 54 		this.setIO = function setIO(io) {
 55 			if (io && typeof io.connect == "function") {
 56 				_io = io;
 57 				return true;
 58 			} else {
 59 				return false;
 60 			}
 61 		};
 62 
 63 		/**
 64 		 * Get socket.io, for debugging purpose
 65 		 * @private
 66 		 * @param
 67 		 * @returns the handler
 68 		 */
 69 		this.getIO = function getIO() {
 70 			return _io;
 71 		};
 72 
 73 		/**
 74 		 * Connect Transport to an url
 75 		 * @param {Url} url the url to connect Transport to
 76 		 */
 77 		this.connect = function connect(url) {
 78 			if (typeof url == "string") {
 79 				_socket = _io.connect(url);
 80 				return true;
 81 			} else {
 82 				return false;
 83 			}
 84 		},
 85 
 86 		/**
 87 		 * Get the socket, for debugging purpose
 88 		 * @private
 89 		 * @returns {Object} the socket
 90 		 */
 91 		this.getSocket = function getSocket() {
 92 			return _socket;
 93 		},
 94 
 95 		/**
 96 		 * Subscribe to a socket event
 97 		 * @param {String} event the name of the event
 98 		 * @param {Function} func the function to execute when the event fires
 99 		 */
100 		this.on = function on(event, func) {
101 			return _socket.on(event, func);
102 		},
103 
104 		/**
105 		 * Subscribe to a socket event but disconnect as soon as it fires.
106 		 * @param {String} event the name of the event
107 		 * @param {Function} func the function to execute when the event fires
108 		 */
109 		this.once = function once(event, func) {
110 			return _socket.once(event, func);
111 		};
112 
113 		/**
114 		 * Publish an event on the socket
115 		 * @param {String} event the event to publish
116 		 * @param data
117 		 * @param {Function} callback is the function to be called for ack
118 		 */
119 		this.emit = function emit(event, data, callback) {
120 			return _socket.emit(event, data, callback);
121 		};
122 
123 		/**
124 		 * Stop listening to events on a channel
125 		 * @param {String} event the event to publish
126 		 * @param data
127 		 * @param {Function} callback is the function to be called for ack
128 		 */
129 		this.removeListener = function emit(event, data, callback) {
130 			return _socket.removeListener(event, data, callback);
131 		};
132 
133 		/**
134 		 * Make a request on the node server
135 		 * @param {String} channel watch the server's documentation to see available channels
136 		 * @param data the request data, it could be anything
137 		 * @param {Function} func the callback that will get the response.
138 		 * @param {Object} scope the scope in which to execute the callback
139 		 */
140 		this.request = function request(channel, data, func, scope) {
141 			if (typeof channel == "string"
142 					&& typeof data != "undefined") {
143 
144 				var reqData = {
145 						eventId: Date.now() + Math.floor(Math.random()*1e6),
146 						data: data
147 					},
148 					boundCallback = function () {
149 						func && func.apply(scope || null, arguments);
150 					};
151 
152 				this.once(reqData.eventId, boundCallback);
153 
154 				this.emit(channel, reqData);
155 
156 				return true;
157 			} else {
158 				return false;
159 			}
160 		};
161 
162 		/**
163 		 * Listen to an url and get notified on new data
164 		 * @param {String} channel watch the server's documentation to see available channels
165 		 * @param data the request data, it could be anything
166 		 * @param {Function} func the callback that will get the data
167 		 * @param {Object} scope the scope in which to execute the callback
168 		 * @returns
169 		 */
170 		this.listen = function listen(channel, data, func, scope) {
171 			if (typeof channel == "string"
172 					&& typeof data != "undefined"
173 					&& typeof func == "function") {
174 
175 				var reqData = {
176 						eventId: Date.now() + Math.floor(Math.random()*1e6),
177 						data: data,
178 						keepAlive: true
179 					},
180 					boundCallback = function () {
181 						func && func.apply(scope || null, arguments);
182 					},
183 					that = this;
184 
185 				this.on(reqData.eventId, boundCallback);
186 
187 				this.emit(channel, reqData);
188 
189 				return function stop() {
190 					that.emit("disconnect-" + reqData.eventId);
191 					that.removeListener(reqData.eventId, boundCallback);
192 				};
193 			} else {
194 				return false;
195 			}
196 		};
197 
198 		/**
199 		 * Initializes the transport to the given url.
200 		 */
201 		this.setIO($io);
202 		this.connect($url);
203 	};
204 });
