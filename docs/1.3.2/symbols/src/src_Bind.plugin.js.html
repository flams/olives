  9 /**
 10  * @class
 11  * This plugin links dom nodes to a model
 12  * @requires Store, Observable
 13  */
 14 function ModelPlugin(Store, Observable, Tools, DomUtils) {
 15 
 16 	return function ModelPluginConstructor($model, $bindings) {
 17 
 18 		/**
 19 		 * The model to watch
 20 		 * @private
 21 		 */
 22 		var _model = null,
 23 
 24 		/**
 25 		 * The list of custom bindings
 26 		 * @private
 27 		 */
 28 		_bindings = {},
 29 
 30 		/**
 31 		 * The list of itemRenderers
 32 		 * each foreach has its itemRenderer
 33 		 * @private
 34 		 */
 35 		_itemRenderers = {};
 36 
 37 		/**
 38 		 * The observers handlers
 39 		 * for debugging only
 40 		 * @private
 41 		 */
 42 		this.observers = {};
 43 
 44 		/**
 45 		 * Define the model to watch for
 46 		 * @param {Store} model the model to watch for changes
 47 		 * @returns {Boolean} true if the model was set
 48 		 */
 49 		this.setModel = function setModel(model) {
 50 			if (model instanceof Store) {
 51 				// Set the model
 52 				_model = model;
 53 				return true;
 54 			} else {
 55 				return false;
 56 			}
 57 		};

 59 		/**
 60 		 * Get the store that is watched for
 61 		 * for debugging only
 62 		 * @private
 63 		 * @returns the Store
 64 		 */
 65 		this.getModel = function getModel() {
 66 			return _model;
 67 		};
 68 
 69 		/**
 70 		 * The item renderer defines a dom node that can be duplicated
 71 		 * It is made available for debugging purpose, don't use it
 72 		 * @private
 73 		 */
 74 		this.ItemRenderer = function ItemRenderer($plugins, $rootNode) {
 75 
 76 			/**
 77 			 * The node that will be cloned
 78 			 * @private
 79 			 */
 80 			var _node = null,
 81 
 82 			/**
 83 			 * The object that contains plugins.name and plugins.apply
 84 			 * @private
 85 			 */
 86 			_plugins = null,
 87 
 88 			/**
 89 			 * The _rootNode where to append the created items
 90 			 * @private
 91 			 */
 92 			_rootNode = null,
 93 
 94 			/**
 95 			 * The lower boundary
 96 			 * @private
 97 			 */
 98 			_start = null,
 99 
100 			/**
101 			 * The number of item to display
102 			 * @private
103 			 */
104 			_nb = null;
105 
106 			/**
107 			 * Set the duplicated node
108 			 * @private
109 			 */
110 			this.setRenderer = function setRenderer(node) {
111 				_node = node;
112 				return true;
113 			};
114 
115 			/**
116 			 * Returns the node that is going to be used for rendering
117 			 * @private
118 			 * @returns the node that is duplicated
119 			 */
120 			this.getRenderer = function getRenderer() {
121 				return _node;
122 			};
123 
124 			/**
125 			 * Sets the rootNode and gets the node to copy
126 			 * @private
127 			 * @param {HTMLElement|SVGElement} rootNode
128 			 * @returns
129 			 */
130 			this.setRootNode = function setRootNode(rootNode) {
131 				var renderer;
132 				if (DomUtils.isAcceptedType(rootNode)) {
133 					_rootNode = rootNode;
134 					renderer = _rootNode.querySelector("*");
135 					this.setRenderer(renderer);
136 					renderer && _rootNode.removeChild(renderer);
137 					return true;
138 				} else {
139 					return false;
140 				}
141 			};
142 
143 			/**
144 			 * Gets the rootNode
145 			 * @private
146 			 * @returns _rootNode
147 			 */
148 			this.getRootNode = function getRootNode() {
149 				return _rootNode;
150 			};
151 
152 			/**
153 			 * Set the plugins objet that contains the name and the apply function
154 			 * @private
155 			 * @param plugins
156 			 * @returns true
157 			 */
158 			this.setPlugins = function setPlugins(plugins) {
159 				_plugins = plugins;
160 				return true;
161 			};
162 
163 			/**
164 			 * Get the plugins object
165 			 * @private
166 			 * @returns the plugins object
167 			 */
168 			this.getPlugins = function getPlugins() {
169 				return _plugins;
170 			};
171 
172 			/**
173 			 * The nodes created from the items are stored here
174 			 * @private
175 			 */
176 			this.items = new Store([]);
177 
178 			/**
179 			 * Set the start limit
180 			 * @private
181 			 * @param {Number} start the value to start rendering the items from
182 			 * @returns the value
183 			 */
184 			this.setStart = function setStart(start) {
185 				return _start = parseInt(start, 10);
186 			};
187 
188 			/**
189 			 * Get the start value
190 			 * @private
191 			 * @returns the start value
192 			 */
193 			this.getStart = function getStart() {
194 				return _start;
195 			};
196 
197 			/**
198 			 * Set the number of item to display
199 			 * @private
200 			 * @param {Number/String} nb the number of item to display or "*" for all
201 			 * @returns the value
202 			 */
203 			this.setNb = function setNb(nb) {
204 				return _nb = nb == "*" ? nb : parseInt(nb, 10);
205 			};
206 
207 			/**
208 			 * Get the number of item to display
209 			 * @private
210 			 * @returns the value
211 			 */
212 			this.getNb = function getNb() {
213 				return _nb;
214 			};
215 
216 			/**
217 			 * Adds a new item and adds it in the items list
218 			 * @private
219 			 * @param {Number} id the id of the item
220 			 * @returns
221 			 */
222 			this.addItem = function addItem(id) {
223 				var node,
224 					next;
225 
226 				if (typeof id == "number" && !this.items.get(id)) {
227 					node = this.create(id);
228 					if (node) {
229 						// IE (until 9) apparently fails to appendChild when insertBefore's second argument is null, hence this.
230 						next = this.getNextItem(id);
231 						next ? _rootNode.insertBefore(node, next) : _rootNode.appendChild(node);
232 						return true;
233 					} else {
234 						return false;
235 					}
236 				} else {
237 					return false;
238 				}
239 			};
240 
241 			/**
242 			 * Get the next item in the item store given an id.
243 			 * @private
244 			 * @param {Number} id the id to start from
245 			 * @returns
246 			 */
247 			this.getNextItem = function getNextItem(id) {
248 				return this.items.alter("slice", id+1).filter(function (value) {
249 					if (DomUtils.isAcceptedType(value)) {
250 						return true;
251 					}
252 				})[0];
253 			};
254 
255 			/**
256 			 * Remove an item from the dom and the items list
257 			 * @private
258 			 * @param {Number} id the id of the item to remove
259 			 * @returns
260 			 */
261 			this.removeItem = function removeItem(id) {
262 				var item = this.items.get(id);
263 				if (item) {
264 					_rootNode.removeChild(item);
265 					this.items.set(id);
266 					return true;
267 				} else {
268 					return false;
269 				}
270 			};
271 
272 			/**
273 			 * create a new node. Actually makes a clone of the initial one
274 			 * and adds pluginname_id to each node, then calls plugins.apply to apply all plugins
275 			 * @private
276 			 * @param id
277 			 * @param pluginName
278 			 * @returns the associated node
279 			 */
280 			this.create = function create(id) {
281 				if (_model.has(id)) {
282 					var newNode = _node.cloneNode(true),
283 					nodes = DomUtils.getNodes(newNode);
284 
285 					Tools.toArray(nodes).forEach(function (child) {
286 	            		child.setAttribute("data-" + _plugins.name+"_id", id);
287 					});
288 
289 					this.items.set(id, newNode);
290 					_plugins.apply(newNode);
291 					return newNode;
292 				}
293 			};
294 
295 			/**
296 			 * Renders the dom tree, adds nodes that are in the boundaries
297 			 * and removes the others
298 			 * @private
299 			 * @returns true boundaries are set
300 			 */
301 			this.render = function render() {
302 				// If the number of items to render is all (*)
303 				// Then get the number of items
304 				var _tmpNb = _nb == "*" ? _model.getNbItems() : _nb;
305 
306 				// This will store the items to remove
307 				var marked = [];
308 
309 				// Render only if boundaries have been set
310 				if (_nb !== null && _start !== null) {
311 
312 					// Loop through the existing items
313 					this.items.loop(function (value, idx) {
314 						// If an item is out of the boundary
315 						if (idx < _start || idx >= (_start + _tmpNb) || !_model.has(idx)) {
316 							// Mark it
317 							marked.push(idx);
318 						}
319 					}, this);
320 
321 					// Remove the marked item from the highest id to the lowest
322 					// Doing this will avoid the id change during removal
323 					// (removing id 2 will make id 3 becoming 2)
324 					marked.sort(Tools.compareNumbers).reverse().forEach(this.removeItem, this);
325 
326 					// Now that we have removed the old nodes
327 					// Add the missing one
328 					for (var i=_start, l=_tmpNb+_start; i<l; i++) {
329 						this.addItem(i);
330 					}
331 					return true;
332 				} else {
333 					return false;
334 				}
335 			};
336 
337 			this.setPlugins($plugins);
338 			this.setRootNode($rootNode);
339 		};
340 
341 		/**
342 		 * Save an itemRenderer according to its id
343 		 * @private
344 		 * @param {String} id the id of the itemRenderer
345 		 * @param {ItemRenderer} itemRenderer an itemRenderer object
346 		 */
347 		this.setItemRenderer = function setItemRenderer(id, itemRenderer) {
348 			id = id || "default";
349 			_itemRenderers[id] = itemRenderer;
350 		};
351 
352 		/**
353 		 * Get an itemRenderer
354 		 * @private
355 		 * @param {String} id the name of the itemRenderer
356 		 * @returns the itemRenderer
357 		 */
358 		this.getItemRenderer = function getItemRenderer(id) {
359 			return _itemRenderers[id];
360 		};
361 
362 		/**
363 		 * Expands the inner dom nodes of a given dom node, filling it with model's values
364 		 * @param {HTMLElement|SVGElement} node the dom node to apply foreach to
365 		 */
366 		this.foreach = function foreach(node, idItemRenderer, start, nb) {
367 			var itemRenderer = new this.ItemRenderer(this.plugins, node);
368 
369 			itemRenderer.setStart(start || 0);
370 			itemRenderer.setNb(nb || "*");
371 
372 			itemRenderer.render();
373 
374 			// Add the newly created item
375             _model.watch("added", itemRenderer.render, itemRenderer);
376 
377 			// If an item is deleted
378             _model.watch("deleted", function (idx) {
379             	itemRenderer.render();
380                 // Also remove all observers
381                 this.observers[idx] && this.observers[idx].forEach(function (handler) {
382                 	_model.unwatchValue(handler);
383                 }, this);
384                 delete this.observers[idx];
385             },this);
386 
387             this.setItemRenderer(idItemRenderer, itemRenderer);
388          };
389 
390          /**
391           * Update the lower boundary of a foreach
392           * @param {String} id the id of the foreach to update
393           * @param {Number} start the new value
394           * @returns true if the foreach exists
395           */
396          this.updateStart = function updateStart(id, start) {
397         	 var itemRenderer = this.getItemRenderer(id);
398         	 if (itemRenderer) {
399         		 itemRenderer.setStart(start);
400         		 return true;
401         	 } else {
402         		 return false;
403         	 }
404          };
405 
406          /**
407           * Update the number of item to display in a foreach
408           * @param {String} id the id of the foreach to update
409           * @param {Number} nb the number of items to display
410           * @returns true if the foreach exists
411           */
412          this.updateNb = function updateNb(id, nb) {
413         	 var itemRenderer = this.getItemRenderer(id);
414         	 if (itemRenderer) {
415         		 itemRenderer.setNb(nb);
416         		 return true;
417         	 } else {
418         		 return false;
419         	 }
420          };

423           * Refresh a foreach after having modified its limits
424           * @param {String} id the id of the foreach to refresh
425           * @returns true if the foreach exists
426           */
427          this.refresh = function refresh(id) {
428         	var itemRenderer = this.getItemRenderer(id);
429     	 	if (itemRenderer) {
430     	 		itemRenderer.render();
431     	 		return true;
432     	 	} else {
433     	 		return false;
434     	 	}
435          };

437 		/**
438 		 * Both ways binding between a dom node attributes and the model
439 		 * @param {HTMLElement|SVGElement} node the dom node to apply the plugin to
440 		 * @param {String} name the name of the property to look for in the model's value
441 		 * @returns
442 		 */
443 		this.bind = function bind(node, property, name) {
444 
445 			// Name can be unset if the value of a row is plain text
446 			name = name || "";
447 
448 			// In case of an array-like model the id is the index of the model's item to look for.
449 			// The _id is added by the foreach function
450 			var id = node.getAttribute("data-" + this.plugins.name+"_id"),
451 
452 			// Else, it is the first element of the following
453 			split = name.split("."),
454 
455 			// So the index of the model is either id or the first element of split
456 			modelIdx = id || split.shift(),
457 
458 			// And the name of the property to look for in the value is
459 			prop = id ? name : split.join("."),
460 
461 			// Get the model's value
462 			get =  Tools.getNestedProperty(_model.get(modelIdx), prop),
463 
464 			// When calling bind like bind:newBinding,param1, param2... we need to get them
465 			extraParam = Tools.toArray(arguments).slice(3);
466 
467 			// 0 and false are acceptable falsy values
468 			if (get || get === 0 || get === false) {
469 				// If the binding hasn't been overriden
470 				if (!this.execBinding.apply(this,
471 						[node, property, get]
472 					// Extra params are passed to the new binding too
473 						.concat(extraParam))) {
474 					// Execute the default one which is a simple assignation
475 					//node[property] = get;
476 					DomUtils.setAttribute(node, property, get);
477 				}
478 			}
479 
480 			// Only watch for changes (double way data binding) if the binding
481 			// has not been redefined
482 			if (!this.hasBinding(property)) {
483 				node.addEventListener("change", function (event) {
484 					if (_model.has(modelIdx)) {
485 						if (prop) {
486 							_model.update(modelIdx, name, node[property]);
487 						} else {
488 							_model.set(modelIdx, node[property]);
489 						}
490 					}
491 				}, true);
492 
493 			}
494 
495 			// Watch for changes
496 			this.observers[modelIdx] = this.observers[modelIdx] || [];
497 			this.observers[modelIdx].push(_model.watchValue(modelIdx, function (value) {
498 				if (!this.execBinding.apply(this,
499 						[node, property, Tools.getNestedProperty(value, prop)]
500 						// passing extra params too
501 						.concat(extraParam))) {
502 					//node[property] = Tools.getNestedProperty(value, prop);
503 					DomUtils.setAttribute(node, property, Tools.getNestedProperty(value, prop));
504 				}
505 			}, this));
506 
507 		};
508 
509 		/**
510 		 * Set the node's value into the model, the name is the model's property
511 		 * @private
512 		 * @param {HTMLElement|SVGElement} node
513 		 * @returns true if the property is added
514 		 */
515 		this.set = function set(node) {
516 			if (DomUtils.isAcceptedType(node) && node.name) {
517 				_model.set(node.name, node.value);
518 				return true;
519 			} else {
520 				return false;
521 			}
522 		};
523 
524 		/**
525 		 * Prevents the submit and set the model with all form's inputs
526 		 * @param {HTMLFormElement} form
527 		 * @returns true if valid form
528 		 */
529 		this.form = function form(form) {
530 			if (form && form.nodeName == "FORM") {
531 				var that = this;
532 				form.addEventListener("submit", function (event) {
533 					Tools.toArray(form.querySelectorAll("[name]")).forEach(that.set, that);
534 					event.preventDefault();
535 				}, true);
536 				return true;
537 			} else {
538 				return false;
539 			}
540 		};
541 
542 		/**
543 		 * Add a new way to handle a binding
544 		 * @param {String} name of the binding
545 		 * @param {Function} binding the function to handle the binding
546 		 * @returns
547 		 */
548 		this.addBinding = function addBinding(name, binding) {
549 			if (name && typeof name == "string" && typeof binding == "function") {
550 				_bindings[name] = binding;
551 				return true;
552 			} else {
553 				return false;
554 			}
555 		};
556 
557 		/**
558 		 * Execute a binding
559 		 * Only used by the plugin
560 		 * @private
561 		 * @param {HTMLElement} node the dom node on which to execute the binding
562 		 * @param {String} name the name of the binding
563 		 * @param {Any type} value the value to pass to the function
564 		 * @returns
565 		 */
566 		this.execBinding = function execBinding(node, name) {
567 			if (this.hasBinding(name)) {
568 				_bindings[name].apply(node, Array.prototype.slice.call(arguments, 2));
569 				return true;
570 			} else {
571 				return false;
572 			}
573 		};
574 
575 		/**
576 		 * Check if the binding exists
577 		 * @private
578 		 * @param {String} name the name of the binding
579 		 * @returns
580 		 */
581 		this.hasBinding = function hasBinding(name) {
582 			return _bindings.hasOwnProperty(name);
583 		};
584 
585 		/**
586 		 * Get a binding
587 		 * For debugging only
588 		 * @private
589 		 * @param {String} name the name of the binding
590 		 * @returns
591 		 */
592 		this.getBinding = function getBinding(name) {
593 			return _bindings[name];
594 		};
595 
596 		/**
597 		 * Add multiple binding at once
598 		 * @param {Object} list the list of bindings to add
599 		 * @returns
600 		 */
601 		this.addBindings = function addBindings(list) {
602 			return Tools.loop(list, function (binding, name) {
603 				this.addBinding(name, binding);
604 			}, this);
605 		};
606 
607 		// Inits the model
608 		this.setModel($model);
609 		// Inits bindings
610 		this.addBindings($bindings);
611

613 	};
