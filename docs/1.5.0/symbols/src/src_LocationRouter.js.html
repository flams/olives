  7 define(["Router", "Tools"],
  8 
  9 /**
 10  * @class
 11  * A locationRouter is a router which navigates to the route defined in the URL and updates this URL
 12  * while navigating. It's a subtype of Emily's Router
 13  */
 14 function LocationRouter(Router, Tools) {
 15 
 16     "use strict";
 17 
 18     function LocationRouterConstructor() {
 19 
 20         /**
 21          * The handle on the watch
 22          * @private
 23          */
 24         var _watchHandle;
 25 
 26         /**
 27          * The function that parses the url to determine the route to navigate to.
 28          * It has a default behavior explained below, but can be overriden as long as
 29          * it has the same contract.
 30          * @param {String} hash the hash coming from window.location.has
 31          * @returns {Array} has to return an array with the list of arguments to call
 32          *    navigate with. The first item of the array must be the name of the route.
 33          *
 34          * Example: #album/holiday/2013
 35          *      will navigate to the route "album" and give two arguments "holiday" and "2013"
 36          */
 37         this.parse = function parse(hash) {
 38             return hash.split("#").pop().split("/");
 39         };
 40 
 41         /**
 42          * The function that converts, or serialises the route and its arguments to a valid URL.
 43          * It has a default behavior below, but can be overriden as long as it has the same contract.
 44          * @param {Array} args the list of arguments to serialize
 45          * @returns {String} the serialized arguments to add to the url hashmark
 46          *
 47          * Example:
 48          *      ["album", "holiday", "2013"];
 49          *      will give "album/holiday/2013"
 50          *
 51          */
 52         this.toUrl = function toUrl(args) {
 53             return args.join("/");
 54         };
 55 
 56         /**
 57          * When all the routes and handlers have been defined, start the location router
 58          * so it parses the URL and navigates to the corresponding route.
 59          * It will also start listening to route changes and hashmark changes to navigate.
 60          * While navigating, the hashmark itself will also change to reflect the current route state
 61          */
 62         this.start = function start() {
 63             var parsedHash = this.parse(window.location.hash);
 64             this.navigate.apply(this, parsedHash);
 65             this.bindOnHashChange();
 66             this.bindOnRouteChange();
 67         };
 68 
 69         /**
 70          * Remove the events handler for cleaning.
 71          */
 72         this.destroy = function destroy() {
 73             this.unwatch(_watchHandle);
 74             window.removeEventListener("hashchange", this.boundOnHashChange, true);
 75         };
 76 
 77         /**
 78          * Parse the hash and navigate to the corresponding url
 79          * @private
 80          */
 81         this.onHashChange  = function onHashChange(event) {
 82             var parsedHash = this.parse(event.newUrl.split("#").pop());
 83             this.navigate.apply(this, parsedHash);
 84         };
 85 
 86         /**
 87          * The bound version of onHashChange for add/removeEventListener
 88          * @private
 89          */
 90         this.boundOnHashChange = this.onHashChange.bind(this);
 91 
 92         /**
 93          * Add an event listener to hashchange to navigate to the corresponding route
 94          * when it changes
 95          * @private
 96          */
 97         this.bindOnHashChange = function bindOnHashChange() {
 98             window.addEventListener("hashchange", this.boundOnHashChange, true);
 99         };
100 
101         /**
102          * Watch route change events from the router to update the location
103          * @private
104          */
105         this.bindOnRouteChange = function bindOnRouteChange() {
106             _watchHandle = this.watch("route", this.onRouteChange, this);
107         };
108 
109         /**
110          * The handler for when the route changes
111          * It updates the location
112          * @private
113          */
114         this.onRouteChange = function onRouteChange() {
115             window.location.hash = this.toUrl(Tools.toArray(arguments));
116         };
117 
118     }
119 
120     return function LocationRouterFactory() {
121         LocationRouterConstructor.prototype = new Router();
122         return new LocationRouterConstructor();
123     };
124 
125 });
